44std::shared_ptr<Obj>
45Tuple::nth(Obj *idx, Expr *expr) {
46 switch (idx->type()) {
47 case Type::Int: {
48 auto index = dynamic_cast<Int *>(idx);
49 if (index->value() < 0 || static_cast<size_t>(index->value()) >= this->value().size()) {
50 Err::err_wexpr(expr);
51 std::string msg = "index "+std::to_string(index->value())+" is out of range of length "+std::to_string(this->value().size());
52 throw InterpreterException(msg);
53 }
54 return this->value().at(index->value());
55 } break;
56 default: {
57 std::string msg = "invalid index when accessing value in a tuple";
58 throw InterpreterException(msg);
59 } break;
60 }
61 return nullptr; // unreachable
62}

71std::shared_ptr<Tuple>
72Tuple::filter(Obj *closure, std::shared_ptr<Ctx> &ctx) {
73 Closure *cl = dynamic_cast<Closure *>(closure);
74
75 auto copy = std::make_shared<Tuple>();
76 std::vector<std::shared_ptr<Obj>> keep_values = {};
77
78 for (size_t i = 0; i < m_values.size(); ++i) {
79 std::vector<std::shared_ptr<Obj>> values = {m_values.at(i)};
80 std::shared_ptr<Obj> filter_result = cl->call(values, ctx);
81 assert(filter_result->type() == Type::Bool);
82 if (dynamic_cast<Bool *>(filter_result.get())->boolean())
83 keep_values.push_back(m_values.at(i)->copy());
84 }
85
86 std::for_each(keep_values.begin(), keep_values.end(), [&](auto &v) {copy->m_values.push_back(v);});
87 return copy;
88}

121std::shared_ptr<Obj>
122Tuple::binop(Token *op, Obj *other) {
123 ASSERT_BINOP_COMPAT(this, other, op);
124
125 auto other_tuple = dynamic_cast<Tuple *>(other);
126 switch (op->type()) {
127 case TokenType::Plus: {
128 std::vector<std::shared_ptr<Obj>> values = {};
129 std::for_each(m_values.begin(), m_values.end(), [&](auto &v){values.push_back(v);});
130 std::for_each(other_tuple->value().begin(), other_tuple->value().end(), [&](auto &v){values.push_back(v);});
131 return std::make_shared<Tuple>(values);
132 } break;
133 case TokenType::Double_Equals: {
134 if (m_values.size() != other_tuple->value().size())
135 return std::make_shared<Bool>(false);
136 for (size_t i = 0; i < m_values.size(); ++i) {
137 if (!m_values[i]->eq(other_tuple->value()[i].get()))
138 return std::make_shared<Bool>(false);
139 }
140 return std::make_shared<Bool>(true);
141 } break;
142 default: {
143 Err::err_wtok(op);
144 std::string msg = "invalid binary operator";
145 throw InterpreterException(msg);
146 }
147 }
148 assert(false && "unreachable");
149 return nullptr;
150}
