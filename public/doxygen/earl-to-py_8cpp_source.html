35#define PYSTMT(s, ctx) \
36 do { \
37 tabs(ctx); \
38 ctx.py_src += s "\n"; \
39 } while (0)
40
41#define PYAPPEND(s, ctx) \
42 do { \
43 ctx.py_src += s; \
44 } while (0)
45
46#define PYSTMT_CONS(s, ctx) \
47 do { \
48 tabs(ctx); \
49 ctx.py_src += s+"\n"; \
50 } while (0)

81struct etopy {
82 std::string py_repl;
83 std::vector<std::string> extra_py_params;
84 std::optional<std::string> imports;
85 etopy(std::string py_repl, std::vector<std::string> extra_py_params,
86 std::optional<std::string> imports)

116static void
117add_ctx_py_import(const std::string &py_import, Context &ctx) {
118 for (auto &i : ctx.py_imports) {
119 if (i == py_import)
120 return;
121 }
122 ctx.py_imports.push_back(py_import);

125static std::string
126earl_intrinsic_to_py_intrinsic(const std::string &orig_funccall,
127 const std::string &comma_sep_params_str,
128 Context &ctx) {
129 auto it = earl_to_py_equiv.find(orig_funccall);
130
131 // Function was called without checking first.
132 if (it == earl_to_py_equiv.end()) {
133 std::cerr <<
134 "EARL intrinsic `"+orig_funccall+"` does not have an equivalent Python intrinsic"
135 << std::endl;
136 std::exit(EXIT_FAILURE);
137 }
138
139 const std::string &py_replacement = it->second.py_repl;
140 std::string py_intr_buf = "";
141 bool found_replacement_pos = false;
142
143 for (size_t i = 0; i < py_replacement.size(); ++i) {
144 char ch = py_replacement.at(i);
145 if (ch == PY_INTRINSIC_REPLACEMENT_CODE) {
146 if (found_replacement_pos) {
147 std::cerr <<
148 "earl_intrinsic_to_py_intrinsic: can only have 1 replacement location in Python intrinsic"
149 << std::endl;
150 std::exit(EXIT_FAILURE);
151 }
152 found_replacement_pos = true;
153 py_intr_buf += comma_sep_params_str;
154
155 // If there are extra parameters that Python needs,
156 // insert them.
157 if (it->second.extra_py_params.size() != 0) {
158 // Handle the case where there are no user parameters passed.
159 if (comma_sep_params_str.size() > 0)
160 py_intr_buf += ", ";
161 for (size_t i = 0; i < it->second.extra_py_params.size(); ++i) {
162 py_intr_buf += it->second.extra_py_params.at(i);
163 if (i != it->second.extra_py_params.size() - 1)
164 py_intr_buf += ", ";
165 }
166 }
167 }
168 else
169 py_intr_buf += ch;
170 }
171
172 // There is an import that we need
173 if (it->second.imports.has_value())
174 add_ctx_py_import(it->second.imports.value(), ctx);
175
176 return py_intr_buf;
177}

186 const std::unordered_map<std::string, Intrinsics::IntrinsicFunction> &map =
187 Intrinsics::intrinsic_functions;
188 return map.find(id) != map.end();

210static std::string
211expr_term_strlit_to_py(ExprStrLit *expr, Context &ctx) {
212 std::string pystr = "";
213 for (size_t i = 0; i < expr->m_tok->lexeme().size(); ++i) {
214 char ch = expr->m_tok->lexeme()[i];
215 if (ch == '\n')
216 pystr += "\\n";
217 else if (ch == '\t')
218 pystr += "\\t";
219 else if (ch == '\r')
220 pystr += "\\r";
221 else
222 pystr += ch;
223 }
224 return "\""+pystr+"\"";
225}

237static std::string
238expr_term_funccall_to_py(ExprFuncCall *expr, Context &ctx) {
239 std::string accessor = expr_to_py(expr->m_left.get(), ctx);
240
241 std::string params = "";
242 for (size_t i = 0; i < expr->m_params.size(); ++i) {
243 params += expr_to_py(expr->m_params.at(i).get(), ctx);
244 if (i != expr->m_params.size() - 1)
245 params += ", ";
246 }
247
248 if (ident_is_intrinsic_function(accessor)) {
249 accessor = earl_intrinsic_to_py_intrinsic(accessor, params, ctx);
250 return accessor;
251 }
252
253 return accessor+"("+params+")";
254}

267static std::string
268expr_term_get_to_py(ExprGet *expr, Context &ctx) {
269 std::string left = expr_to_py(expr->m_left.get(), ctx);
270 std::string right = "";
271
272 std::visit([&](auto &&arg) {
273 using T = std::decay_t<decltype(arg)>;
274 if constexpr (std::is_same_v<T, std::unique_ptr<ExprIdent>>)
275 right = expr_to_py(arg.get(), ctx);
276 else if constexpr (std::is_same_v<T, std::unique_ptr<ExprFuncCall>>)
277 right = expr_to_py(arg.get(), ctx);
278 else {
279 std::string msg = "A serious internal error has ocured and has gotten to an unreachable case. Something is very wrong";
280 throw InterpreterException(msg);
281 }
282 }, expr->m_right);
283
284 std::string ident = "";
285 std::string new_right = "";
286 size_t open_paren = 0;
287 size_t close_paren = 0;
288 for (size_t i = 0; i < right.size(); ++i) {
289 if (right.at(i) == '(') {
290 for (size_t j = i; j < right.size(); ++j)
291 new_right += right.at(j);
292 break;
293 }
294 ident += right.at(i);
295 }
296
297 if (Intrinsics::intrinsic_member_functions.find(ident) !=
298 Intrinsics::intrinsic_member_functions.end()) {
299 ctx.unresolved_idents.push_back(ident);
300 }
301
302 for (size_t i = 0; i < ctx.enum_ids.size(); ++i) {
303 if (left == ctx.enum_ids.at(i))
304 return right;
305 }
306
307 return left+"."+right;
308}

310static std::string
311expr_term_mod_to_py_access(ExprModAccess *expr, Context &ctx) {
312 const std::string pyleft = expr_to_py(expr->m_expr_ident.get(), ctx);
313 std::string pyright = "";
314 std::visit([&](auto &&arg) {
315 using T = std::decay_t<decltype(arg)>;
316 if constexpr (std::is_same_v<T, std::unique_ptr<ExprIdent>>)
317 pyright = expr_to_py(arg.get(), ctx);
318 else if constexpr (std::is_same_v<T, std::unique_ptr<ExprFuncCall>>)
319 pyright = expr_to_py(arg.get(), ctx);
320 else {
321 std::string msg = "A serious internal error has ocured and has gotten to an unreachable case. Something is very wrong";
322 throw InterpreterException(msg);
323 }
324 }, expr->m_right);
325
326 return pyleft+"."+pyright;
327}

346static std::string construct_closure(ExprClosure *expr, Context &ctx) {
347 std::string lambda_name = "__EARL_CLOSURE__" + std::to_string(lambdas_count++);
348 std::string pylambda = "def " + lambda_name + "(";
349 for (size_t i = 0; i < expr->m_args.size(); ++i) {
350 pylambda += expr->m_args.at(i).first->lexeme();
351 if (i != expr->m_args.size()-1)
352 pylambda += ", ";
353 }
354 pylambda += "):\n";
355
356 Context clctx = {0, pylambda, false, {}, {}, {}, {}, {}};
357 stmt_block_to_py(expr->m_block.get(), clctx);
358
359 ctx.lambdas.push_back(clctx.py_src);
360
361 for (size_t i = 0; i < clctx.lambdas.size(); ++i)
362 ctx.lambdas.push_back(clctx.lambdas.at(i));
363
364 return lambda_name;
365}

372static std::string
373expr_term_range_to_py(ExprRange *expr, Context &ctx) {
374 std::string pylst = "[";
375 const std::string pystart = expr_to_py(expr->m_start.get(), ctx);
376 const std::string pyend = expr_to_py(expr->m_end.get(), ctx);
377 bool ischar = false;
378 int start = 0;
379 int end = 0;
380
381 if (pystart[0] == '\'') {
382 start = pystart[1];
383 end = pyend[1];
384 ischar = true;
385 }
386 else {
387 start = std::stoi(pystart);
388 end = std::stoi(pyend);
389 }
390
391 for (int i = start; i < end; ++i) {
392 if (ischar) {
393 pylst += "'";
394 pylst += (char)i;
395 pylst += "'";
396 }
397 else
398 pylst += std::to_string(i);
399 if (i != end - 1)
400 pylst += ", ";
401 }
402
403 if (expr->m_inclusive) {
404 if (ischar) {
405 pylst += ", '";
406 pylst += (char)end;
407 pylst += "'";
408 }
409 else
410 pylst += ", " + std::to_string(end);
411 }
412
413 pylst += "]";
414 return pylst;
415}

459static std::string
460expr_term_to_py(ExprTerm *expr, Context &ctx) {
461 switch (expr->get_term_type()) {
462 case ExprTermType::Ident: return expr_term_ident_to_py(dynamic_cast<ExprIdent *>(expr), ctx);
463 case ExprTermType::Int_Literal: return expr_term_intlit_to_py(dynamic_cast<ExprIntLit *>(expr), ctx);
464 case ExprTermType::Str_Literal: return expr_term_strlit_to_py(dynamic_cast<ExprStrLit *>(expr), ctx);
465 case ExprTermType::Char_Literal: return expr_term_charlit_to_py(dynamic_cast<ExprCharLit *>(expr), ctx);
466 case ExprTermType::Float_Literal: return expr_term_floatlit_to_py(dynamic_cast<ExprFloatLit *>(expr), ctx);
467 case ExprTermType::Func_Call: return expr_term_funccall_to_py(dynamic_cast<ExprFuncCall *>(expr), ctx);
468 case ExprTermType::List_Literal: return expr_term_listlit_to_py(dynamic_cast<ExprListLit *>(expr), ctx);
469 case ExprTermType::Get: return expr_term_get_to_py(dynamic_cast<ExprGet *>(expr), ctx);
470 case ExprTermType::Mod_Access: return expr_term_mod_to_py_access(dynamic_cast<ExprModAccess *>(expr), ctx);
471 case ExprTermType::Array_Access: return expr_term_array_access_to_py(dynamic_cast<ExprArrayAccess *>(expr), ctx);
472 case ExprTermType::Bool: return expr_term_boollit_to_py(dynamic_cast<ExprBool *>(expr), ctx);
473 case ExprTermType::None: return expr_term_none_to_py(dynamic_cast<ExprNone *>(expr), ctx);
474 case ExprTermType::Closure: return expr_term_closure_to_py(dynamic_cast<ExprClosure *>(expr), ctx);
475 case ExprTermType::Range: return expr_term_range_to_py(dynamic_cast<ExprRange *>(expr), ctx);
476 case ExprTermType::Tuple: return expr_term_tuple_to_py(dynamic_cast<ExprTuple *>(expr), ctx);
477 case ExprTermType::Slice: return expr_term_slice_to_py(dynamic_cast<ExprSlice *>(expr), ctx);
478 case ExprTermType::Dict: return expr_term_dict_to_py(dynamic_cast<ExprDict *>(expr), ctx);
479 case ExprTermType::FStr: return expr_term_fstr_to_py(dynamic_cast<ExprFStr *>(expr), ctx);
480 default: {
481 std::string msg = "unknown term: `"+std::to_string((int)expr->get_term_type())+"`";
482 throw InterpreterException(msg);
483 }
484 }
485}

487static std::string
488expr_bin_to_py(ExprBinary *expr, Context &ctx) {
489 std::string lhs = expr_to_py(expr->m_lhs.get(), ctx);
490 std::string rhs = expr_to_py(expr->m_rhs.get(), ctx);
491 std::string op = "";
492 if (expr->m_op->lexeme() == "`&")
493 op = "&";
494 else if (expr->m_op->lexeme() == "`|")
495 op = "|";
496 else if (expr->m_op->lexeme() == "`^")
497 op = "^";
498 else if (expr->m_op->lexeme() == "&&")
499 op = "and";
500 else if (expr->m_op->lexeme() == "||")
501 op = "or";
502 else
503 op = expr->m_op->lexeme();
504 return lhs+" "+op+" "+rhs;
505}

507static std::string
508expr_unary_to_py(ExprUnary *expr, Context &ctx) {
509 std::string pyexpr = expr_to_py(expr->m_expr.get(), ctx);
510 std::string op = "";
511 if (expr->m_op->lexeme() == "`~")
512 op = "~";
513 else if (expr->m_op->lexeme() == "!")
514 op = "not ";
515 else
516 op = expr->m_op->lexeme();
517 return op+pyexpr;
518}

520static std::string
521expr_to_py(Expr *expr, Context &ctx) {
522 switch (expr->get_type()) {
523 case ExprType::Term: {
524 return expr_term_to_py(dynamic_cast<ExprTerm *>(expr), ctx);
525 } break;
526 case ExprType::Binary: {
527 return "("+expr_bin_to_py(dynamic_cast<ExprBinary *>(expr), ctx)+")";
528 } break;
529 case ExprType::Unary: {
530 return expr_unary_to_py(dynamic_cast<ExprUnary *>(expr), ctx);
531 } break;
532 default: assert(false && "unreachable"); break;
533 }
534}

536static void
537stmt_def_to_py(StmtDef *stmt, Context &ctx, bool add_self = false) {
538 const std::string &id = stmt->m_id->lexeme();
539 std::string params = "";
540
541 if (add_self)
542 params += "self";
543
544 if (stmt->m_args.size() > 0 && add_self)
545 params += ", ";
546
547 for (size_t i = 0; i < stmt->m_args.size(); ++i) {
548 params += stmt->m_args.at(i).first.first->lexeme();
549 if (i != stmt->m_args.size() - 1)
550 params += ", ";
551 }
552 PYSTMT_CONS("def "+id+"("+params+"):", ctx);
553 stmt_block_to_py(stmt->m_block.get(), ctx);
554}

556static void
557stmt_let_to_py(StmtLet *stmt, Context &ctx, bool __init__ = false) {
558 std::string pylet = "";
559 for (size_t i = 0; i < stmt->m_ids.size(); ++i) {
560 std::string id = "";
561 if (__init__)
562 pylet += "self.";
563 pylet += stmt->m_ids.at(i)->lexeme();
564 if (i != stmt->m_ids.size()-1)
565 pylet += ", ";
566 }
567
568 const std::string pyexpr = expr_to_py(stmt->m_expr.get(), ctx);
569 PYSTMT_CONS(pylet+" = " + pyexpr, ctx);
570}

572static void
573stmt_block_to_py(StmtBlock *stmt, Context &ctx) {
574 ctx.scope_depth++;
575
576 if (stmt->m_stmts.size() == 0)
577 PYSTMT("pass", ctx);
578
579 for (size_t i = 0; i < stmt->m_stmts.size(); ++i)
580 stmt_to_py(stmt->m_stmts.at(i).get(), ctx);
581
582 ctx.scope_depth--;
583}

698 stmt_let_to_py(stmt->m_members.at(i).get(), ctx, true);

773earl_to_py(std::unique_ptr<Program> program) {
774 if (!program) {
775 return "";
776 }
777
778 Context ctx = {0, "", false, {}, {}, {}, {}, {}};
779 for (size_t i = 0; i < program->m_stmts.size(); ++i)
780 stmt_to_py(program->m_stmts.at(i).get(), ctx);
781
782 std::string unresolved = "";
783 if (ctx.unresolved_idents.size() > 0)
784 unresolved += "# UNRESOLVED SYMBOLS\n";
785 for (auto &l : ctx.unresolved_idents)

787
788 std::string imports = "";
789 if (ctx.earl_imports.size() > 0)
790 imports += "# EARL Imports\n";
791 for (auto &l : ctx.earl_imports)
792 imports += l + "\n";
793
794 std::string py_imports = "";
795 if (ctx.py_imports.size() > 0)
796 py_imports += "# Python Imports\n";
797 for (auto &l : ctx.py_imports)
798 py_imports += "import "+l+"\n";
799
800 std::string lambdas = "";
801 if (ctx.lambdas.size() > 0)
802 lambdas += "# Autogenerated Lambdas\n";
803 for (auto &l : ctx.lambdas)
804 lambdas += l + "\n";
805

808 enum_values += "# Substitute Enum Values\n";
809 for (auto &l : ctx.enum_values)

811

815 + lambdas
816 + enum_values
