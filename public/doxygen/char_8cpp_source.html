95std::shared_ptr<Obj>
96Char::equality(Token *op, Obj *other) {
97 ASSERT_BINOP_COMPAT(this, other, op);
98 switch (op->type()) {
99 case TokenType::Double_Equals: {
100 if (other->type() == Type::Str) {
101 auto str = dynamic_cast<Str *>(other);
102 if (str->value().size() != 1)
103 return std::make_shared<Bool>(false);
104 return std::make_shared<Bool>(this->value() == str->value()[0]);
105 }
106 return std::make_shared<Bool>(this->value() == dynamic_cast<Char *>(other)->value());
107 } break;
108 case TokenType::Bang_Equals: {
109 if (other->type() == Type::Str) {
110 auto str = dynamic_cast<Str *>(other);
111 if (str->value().size() != 1)
112 return std::make_shared<Bool>(true);
113 return std::make_shared<Bool>(this->value() != str->value()[0]);
114 }
115 return std::make_shared<Bool>(this->value() != dynamic_cast<Char *>(other)->value());
116 } break;
117 default: {
118 Err::err_wtok(op);
119 std::string msg = "invalid binary operator";
120 throw InterpreterException(msg);
121 } break;
122 }
123 return nullptr; // unreachable
124}
