119std::shared_ptr<Token>
120token_alloc(Lexer &lexer, char *start, size_t len, TokenType type, size_t row, size_t col, std::string fp) {
121 (void)lexer;
122 if (type == TokenType::Strlit || type == TokenType::Bashlit) {
123 std::string s = "";
124 for (size_t i = 0; i < len; ++i) {
125 if (*(start+i) == '\\' && *(start+i+1) && *(start+i+1) == 'n') {
126 s += '\n';
127 ++i;
128 }
129 else if (*(start+i) == '\\' && *(start+i+1) && *(start+i+1) == 'r') {
130 s += '\r';
131 ++i;
132 }
133 else if (*(start+i) == '\\' && *(start+i+1) && *(start+i+1) == 't') {
134 s += '\t';
135 ++i;
136 }
137 else if (*(start+i) == '\\' && *(start+i+1) && *(start+i+1) == '"') {
138 s += '"';
139 ++i;
140 }
141 else if (*(start+i) == '\\' && *(start+i+1) && *(start+i+1) == '\\') {
142 s += '\\';
143 ++i;
144 }
145 else if (*(start+i) == '\\' && *(start+i+1)) {
146 auto err = std::make_shared<Token>(start, len, type, row, col, fp);
147 const std::string msg = "unknown escape sequence: `\\" + std::string(1, *(start+i+1));
148 Err::err_wtok(err.get());
149 throw ParserException(msg);
150 }
151 else {
152 s += *(start+i);
153 }
154 }
155 return std::make_shared<Token>(s, type, row, col, fp);
156 }
157 return std::make_shared<Token>(start, len, type, row, col, fp);
158}

170void
171token_dump_until_eol(Token *tok, int padding) {
172 for (int i = 0; i < padding; ++i)
173 std::cout << ' ';
174 while (tok && tok->type() != TokenType::Semicolon) {
175 std::cout << tok->lexeme();
176
177 if (tok->m_next && tok->m_next->type() != TokenType::Semicolon) {
178 std::cout << ' ';
179 }
180
181 tok = tok->m_next.get();
182 }
183
184 std::cout << std::endl;
185}
