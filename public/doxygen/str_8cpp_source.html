59std::shared_ptr<Char>
60Str::nth(Obj *idx, Expr *expr) {
61 if (idx->type() != Type::Int) {
62 Err::err_wexpr(expr);
63 std::string msg = "invalid index when accessing value in a str";
64 throw InterpreterException(msg);
65 }
66
67 auto index = dynamic_cast<Int *>(idx);
68 int I = index->value();
69 if (I < 0 || static_cast<size_t>(I) >= m_value.size()) {
70 Err::err_wexpr(expr);
71 std::string msg = "index "+std::to_string(index->value())+" is out of str range of length "+std::to_string(this->value().size());
72 throw InterpreterException(msg);
73 }
74
75 if (m_chars.at(I)) {
76 m_value.at(I) = m_chars.at(I)->value();
77 return m_chars.at(I);
78 }
79
80 auto c = std::make_shared<Char>(m_value.at(I));
81 m_chars.at(I) = std::move(c);
82 m_changed.push_back(I);
83
84 return m_chars.at(I);
85}

88std::shared_ptr<List>
89Str::split(Obj *delim, Expr *expr) {
90 if (delim->type() != Type::Str) {
91 Err::err_wexpr(expr);
92 const std::string msg = "cannot use member intrinsic `split` with non-str type";
93 throw InterpreterException(msg);
94 }
95
96 this->update_changed();
97
98 std::vector<std::shared_ptr<Obj>> splits = {};
99 std::string delim_str = dynamic_cast<Str *>(delim)->value();
100 std::string::size_type start = 0;
101
102 auto pos = this->value().find(delim_str);
103 std::string orig_value = this->value();
104
105 while (pos != std::string::npos) {
106 splits.push_back(std::make_shared<Str>(orig_value.substr(start, pos-start)));
107 start = pos+delim_str.length();
108 pos = orig_value.find(delim_str, start);
109 }
110 splits.push_back(std::make_shared<Str>(orig_value.substr(start)));
111
112 return std::make_shared<List>(std::move(splits));
113}

133 (void)expr;
134 auto *idx1 = dynamic_cast<earl::value::Int *>(idx);
135 int I = idx1->value();
136
137 if (I >= m_value.size()) {
138 Err::err_wexpr(expr);

142
143 this->update_changed();

145 m_chars.erase(m_chars.begin() + I);

215std::shared_ptr<Str>
216Str::filter(Obj *closure, std::shared_ptr<Ctx> &ctx) {
217 this->update_changed();
218
219 Closure *cl = dynamic_cast<Closure *>(closure);
220
221 auto acc = std::make_shared<Str>();
222
223 for (int i = 0; i < m_value.size(); ++i) {
224 std::shared_ptr<Char> cx = nullptr;
225 if (m_chars.at(i)) {
226 m_value.at(i) = m_chars.at(i)->value();
227 cx = m_chars.at(i);
228 }
229 else {
230 auto tmpc = std::make_shared<Char>(m_value.at(i));
231 m_chars.at(i) = std::move(tmpc);
232 cx = m_chars.at(i);
233 m_changed.push_back(i);
234 }
235 std::vector<std::shared_ptr<Obj>> values = {cx};
236 std::shared_ptr<Obj> filter_result = cl->call(values, ctx);
237 if (dynamic_cast<Bool *>(filter_result.get())->boolean())
238 acc->append(cx.get());
239 }
240
241 return acc;
242}

245Str::contains(Char *value) {
246 for (size_t i = 0; i < m_value.size(); ++i) {
247 if (m_chars.at(i) && m_chars.at(i)->value() != m_value.at(i))
248 m_value.at(i) = m_chars.at(i)->value();
249
250 if (m_value.at(i) == value->value())
251 return std::make_shared<Bool>(true);
252 }
253 return std::make_shared<Bool>(false);

282std::shared_ptr<Obj>
283Str::binop(Token *op, Obj *other) {
284 ASSERT_BINOP_COMPAT(this, other, op);
285 this->update_changed();
286 switch (op->type()) {
287 case TokenType::Plus: {
288 return std::make_shared<Str>(this->value() + dynamic_cast<Str *>(other)->value());
289 } break;
290 case TokenType::Double_Equals: {
291 return std::make_shared<Bool>(this->value() == dynamic_cast<Str *>(other)->value());
292 } break;
293 case TokenType::Bang_Equals: {
294 return std::make_shared<Bool>(this->value() != dynamic_cast<Str *>(other)->value());
295 } break;
296 default: {
297 Err::err_wtok(op);
298 std::string msg = "invalid binary operator";
299 throw InterpreterException(msg);
300 }
301 }
302}

404void
405Str::iter_next(Iterator &it) {
406 std::visit([&](auto &iter) {
407 using IteratorType = std::decay_t<decltype(iter)>;
408
409 if constexpr (std::is_same_v<IteratorType, std::vector<std::shared_ptr<Char>>::iterator>) {
410 std::advance(iter, 1);
411
412 if (iter == m_chars.end())
413 return;
414
415 size_t index = std::distance(m_chars.begin(), iter);
416
417 if (index < m_value.size()) {
418 auto c = std::make_shared<Char>(m_value.at(index));
419 *iter = std::move(c);
420 m_changed.push_back(index);
421 }
422 }
423 // No else case needed since we only handle std::vector<std::shared_ptr<Char>>::iterator
424 }, it);
425}

435std::shared_ptr<Obj>
436Str::equality(Token *op, Obj *other) {
437 ASSERT_BINOP_COMPAT(this, other, op);
438 switch (op->type()) {
439 case TokenType::Double_Equals: {
440 if (other->type() == Type::Char) {
441 auto ch = dynamic_cast<Char *>(other);
442 if (this->value().size() != 1)
443 return std::make_shared<Bool>(false);
444 return std::make_shared<Bool>(this->value()[0] == ch->value());
445 }
446 return std::make_shared<Bool>(this->value() == dynamic_cast<Str *>(other)->value());
447 } break;
448 case TokenType::Bang_Equals: {
449 if (other->type() == Type::Char) {
450 auto ch = dynamic_cast<Char *>(other);
451 if (this->value().size() != 1)
452 return std::make_shared<Bool>(true);
453 return std::make_shared<Bool>(this->value()[0] != ch->value());
454 }
455 return std::make_shared<Bool>(this->value() != dynamic_cast<Str *>(other)->value());
456 } break;
457 default: {
458 Err::err_wtok(op);
459 std::string msg = "invalid binary operator";
460 throw InterpreterException(msg);
461 }
462 }
463 return nullptr; // unreachable
