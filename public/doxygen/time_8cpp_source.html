127std::shared_ptr<Obj>
128Time::gtequality(Token *op, Obj *other) {
129 ASSERT_BINOP_COMPAT(this, other, op);
130 auto time2 = dynamic_cast<Time *>(other);
131 switch (op->type()) {
132 case TokenType::Lessthan: return std::make_shared<Bool>(m_now < time2->m_now);
133 case TokenType::Greaterthan: return std::make_shared<Bool>(m_now > time2->m_now);
134 case TokenType::Greaterthan_Equals: return std::make_shared<Bool>(m_now >= time2->m_now);
135 case TokenType::Lessthan_Equals: return std::make_shared<Bool>(m_now <= time2->m_now);
136 default: {
137 Err::err_wtok(op);
138 const std::string msg = "invalid operator";
139 throw InterpreterException(msg);
140 } break;
141 }
142}
