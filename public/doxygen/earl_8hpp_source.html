38#define ASSERT_BINOP_COMPAT(obj0, obj1, op) \
39 do { \
40 if (!type_is_compatable(obj0, obj1)) { \
41 Err::err_wtok(op); \
42 std::string __Msg = "value of type `"+earl::value::type_to_str(obj0->type()) \
43 +"` is not compatible with value of type `"+earl::value::type_to_str(obj1->type())+"`"; \
44 throw InterpreterException(__Msg); \
45 } \
46 } while (0)
47
48#define ASSERT_BINOP_EXACT(obj0, obj1, op) \
49 do { \
50 if (obj0->type() != obj1->type()) { \
51 Err::err_wtok(op); \
52 std::string __Msg = "value of type `"+earl::value::type_to_str(obj0->type()) \
53 +"` is not exactly compatible with value of type `"+earl::value::type_to_str(obj1->type())+"`"; \
54 throw InterpreterException(__Msg); \
55 } \
56 } while (0)
57
58#define ASSERT_MUTATE_COMPAT(obj0, obj1, stmt) \
59 do { \
60 if (!type_is_compatable(obj0, obj1)) { \
61 Err::err_wstmt(stmt); \
62 std::string __Msg = "value of type `"+earl::value::type_to_str(obj0->type()) \
63 +"` is not compatible with value of type `"+earl::value::type_to_str(obj1->type())+"`"; \
64 throw InterpreterException(__Msg); \
65 } \
66 } while (0)
67
68#define ASSERT_CONSTNESS(obj, stmt) \
69 do { \
70 if (obj->is_const()) { \
71 Err::err_wstmt(stmt); \
72 const std::string __Msg = "cannot mutate value with attribute @const"; \
73 throw InterpreterException(__Msg); \
74 } \
75 } while (0)

951 return std::make_shared<earl::value::Option>(value->second);
952 }
953 else if constexpr (std::is_same_v<T, double>) {
954 if (key->type() != earl::value::Type::Float) {
955 Err::err_wexpr(expr);
956 const std::string msg = "key must be of type float";
957 throw InterpreterException(msg);

1030template <typename T> std::string
1031earl::value::Dict<T>::to_cxxstring(void) {
1032 std::string res = "<" + earl::value::type_to_str(this->type()) + " { ";
1033 auto map = this->extract();
1034 int i = 0;
1035 for (auto it = map.begin(); it != map.end(); ++it) {
1036 using Tx = std::decay_t<T>;
1037 if constexpr (std::is_same_v<Tx, int>)
1038 res += std::to_string(it->first);
1039 else if constexpr (std::is_same_v<Tx, double>)
1040 res += std::to_string(it->first);
1041 else if constexpr (std::is_same_v<Tx, char>)
1042 res += std::string(1, it->first);
1043 else if constexpr (std::is_same_v<Tx, std::string>)
1044 res += it->first;
1045 else
1046 res += it->first;
1047 res += ": ";
1048 res += it->second->to_cxxstring();
1049 if (i != map.size()-1)
1050 res += ", ";
1051 ++i;
1052 }
1053 res += " }>";
1054 return res;
1055}
