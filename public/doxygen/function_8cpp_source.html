86void
87Obj::load_parameters(std::vector<std::shared_ptr<earl::value::Obj>> &values,
88 std::shared_ptr<FunctionCtx> &new_ctx,
89 std::shared_ptr<Ctx> &old_ctx) {
90 for (size_t i = 0; i < values.size(); ++i) {
91 auto value = values[i];
92 Token *id = m_params.at(i).first.first;
93 __Type *ty = m_params.at(i).first.second;
94
95 if (ty) {
96 Interpreter::typecheck(ty, value.get(), old_ctx);
97 }
98
99 std::shared_ptr<earl::variable::Obj> var = nullptr;
100 if ((m_params.at(i).second & static_cast<uint32_t>(Attr::Ref)) != 0)
101 var = std::make_shared<earl::variable::Obj>(id, value);
102 else
103 var = std::make_shared<earl::variable::Obj>(id, value->copy());
104 if ((m_params.at(i).second & static_cast<uint32_t>(Attr::Const)) != 0)
105 var->value()->set_const();
106 new_ctx->variable_add(var);
107 }
108}
