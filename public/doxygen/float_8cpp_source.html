52std::shared_ptr<Obj>
53Float::binop(Token *op, Obj *other) {
54 ASSERT_BINOP_COMPAT(this, other, op);
55
56 switch (op->type()) {
57 case TokenType::Plus: {
58 return other->type() == Type::Int ?
59 std::make_shared<Float>(this->value() + dynamic_cast<Int *>(other)->value()) :
60 std::make_shared<Float>(this->value() + dynamic_cast<Float *>(other)->value());
61 } break;
62 case TokenType::Minus: {
63 return other->type() == Type::Int ?
64 std::make_shared<Float>(this->value() - dynamic_cast<Int *>(other)->value()) :
65 std::make_shared<Float>(this->value() - dynamic_cast<Float *>(other)->value());
66 } break;
67 case TokenType::Asterisk: {
68 return other->type() == Type::Int ?
69 std::make_shared<Float>(this->value() * dynamic_cast<Int *>(other)->value()) :
70 std::make_shared<Float>(this->value() * dynamic_cast<Float *>(other)->value());
71 } break;
72 case TokenType::Forwardslash: {
73 return other->type() == Type::Int ?
74 std::make_shared<Float>(this->value() / dynamic_cast<Int *>(other)->value()) :
75 std::make_shared<Float>(this->value() / dynamic_cast<Float *>(other)->value());
76 } break;
77 case TokenType::Percent: {
78 std::string msg = "cannot use module `%%` with floating point";
79 throw InterpreterException(msg);
80 } break;
81 case TokenType::Double_Asterisk: {
82 float p = 0.;
83 if (other->type() == earl::value::Type::Float) {
84 auto _other = dynamic_cast<earl::value::Float *>(other);
85 p = std::pow(static_cast<float>(m_value), static_cast<float>(_other->value()));
86 }
87 else {
88 auto _other = dynamic_cast<earl::value::Int *>(other);
89 p = std::pow(static_cast<float>(m_value), static_cast<float>(_other->value()));
90 }
91 return std::make_shared<earl::value::Float>(static_cast<double>(p));
92 } break;
93 case TokenType::Lessthan: {
94 return other->type() == Type::Int ?
95 std::make_shared<Bool>(this->value() < dynamic_cast<Int *>(other)->value()) :
96 std::make_shared<Bool>(this->value() < dynamic_cast<Float *>(other)->value());
97 } break;
98 case TokenType::Greaterthan: {
99 return other->type() == Type::Int ?
100 std::make_shared<Bool>(this->value() > dynamic_cast<Int *>(other)->value()) :
101 std::make_shared<Bool>(this->value() > dynamic_cast<Float *>(other)->value());
102 } break;
103 case TokenType::Double_Equals: {
104 return other->type() == Type::Int ?
105 std::make_shared<Bool>(this->value() == dynamic_cast<Int *>(other)->value()) :
106 std::make_shared<Bool>(this->value() == dynamic_cast<Float *>(other)->value());
107 } break;
108 case TokenType::Greaterthan_Equals: {
109 return other->type() == Type::Int ?
110 std::make_shared<Bool>(this->value() >= dynamic_cast<Int *>(other)->value()) :
111 std::make_shared<Bool>(this->value() >= dynamic_cast<Float *>(other)->value());
112 } break;
113 case TokenType::Lessthan_Equals: {
114 return other->type() == Type::Int ?
115 std::make_shared<Bool>(this->value() <= dynamic_cast<Int *>(other)->value()) :
116 std::make_shared<Bool>(this->value() <= dynamic_cast<Float *>(other)->value());
117 } break;
118 case TokenType::Bang_Equals: {
119 return other->type() == Type::Int ?
120 std::make_shared<Bool>(this->value() != dynamic_cast<Int *>(other)->value()) :
121 std::make_shared<Bool>(this->value() != dynamic_cast<Float *>(other)->value());
122 } break;
123 case TokenType::Double_Pipe: {
124 return other->type() == Type::Int ?
125 std::make_shared<Bool>(this->value() || dynamic_cast<Int *>(other)->value()) :
126 std::make_shared<Bool>(this->value() || dynamic_cast<Float *>(other)->value());
127 } break;
128 default: {
129 Err::err_wtok(op);
130 std::string msg = "invalid binary operator";
131 throw InterpreterException(msg);
132 }
133 }
134}

142Float::mutate(Obj *other, StmtMut *stmt) {
143 ASSERT_MUTATE_COMPAT(this, other, stmt);
144 ASSERT_CONSTNESS(this, stmt);
145
146 switch (other->type()) {
147 case Type::Float: {
148 m_value = dynamic_cast<Float *>(other)->value();

150 case Type::Int: {
151 m_value = static_cast<double>(dynamic_cast<Int *>(other)->value());
152 } break;
153 default: {
154 assert(false && "unreachable");
155 }
156 }

181void
182Float::spec_mutate(Token *op, Obj *other, StmtMut *stmt) {
183 ASSERT_CONSTNESS(this, stmt);
184 double prev;
185
186 if (op->type() == TokenType::Minus_Equals
187 || op->type() == TokenType::Forwardslash_Equals
188 || op->type() == TokenType::Percent_Equals) {
189 if (other->type() == Type::Int)
190 prev = static_cast<int>(dynamic_cast<Int *>(other)->value());
191 else if (other->type() == Type::Float)
192 prev = dynamic_cast<Float *>(other)->value();
193 else {
194 const std::string msg = "invalid type for spec_mutate";
195 Err::err_wtok(op);
196 throw InterpreterException(msg);
197 }
198 }
199 else {
200 prev = m_value;
201 this->mutate(other, stmt); // does type checking
202 }
203
204 this->mutate(other, stmt); // does type checking
205 switch (op->type()) {
206 case TokenType::Plus_Equals: m_value += prev; break;
207 case TokenType::Minus_Equals: m_value -= prev; break;
208 case TokenType::Asterisk_Equals: m_value *= prev; break;
209 case TokenType::Forwardslash_Equals: m_value /= prev; break;
210 case TokenType::Percent_Equals: {
211 Err::err_wtok(op);
212 std::string msg = "cannot use module `"+op->lexeme()+"` on float type";
213 throw InterpreterException(msg);
214 }
215 default: {
216 Err::err_wtok(op);
217 std::string msg = "invalid operator for special mutation `"+op->lexeme()+"`";
218 throw InterpreterException(msg);
219 } break;
220 }
221}
