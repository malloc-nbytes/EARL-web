74static std::string
75flatten_info(const std::vector<std::string> &lines) {
76 std::string info = "";
77 for (size_t i = 0; i < lines.size(); ++i) {
78 info += lines.at(i);
79 if (i != lines.size()-1)
80 info += '\n';
81 }

85static std::string
86identifier_not_declared(std::string given, std::vector<std::string> possible, bool include_intrinsics=true) {
87 if (include_intrinsics) {
88 for (auto it = Intrinsics::intrinsic_functions.begin(); it != Intrinsics::intrinsic_functions.end(); ++it)
89 possible.push_back(it->first);
90 }
91
92 std::vector<int> ranks = {};
93 int min = 1e9;
94 int min_idx = 0;
95
96 for (size_t i = 0; i < possible.size(); ++i) {
97 int rank = levenshtein_distance(given, possible[i]);
98 if (rank < min) {
99 min = rank;
100 min_idx = i;
101 }
102 }
103
104 return possible.at(min_idx);
105}

107static std::string
108method_not_declared(std::string given, std::shared_ptr<earl::value::Obj> accessor) {
109 std::vector<std::string> possible = {};
110 switch (accessor->type()) {
111 case earl::value::Type::List: {
112 for (auto it = Intrinsics::intrinsic_list_member_functions.begin(); it != Intrinsics::intrinsic_list_member_functions.end(); ++it)
113 possible.push_back(it->first);
114 } break;
115 case earl::value::Type::Str: {
116 for (auto it = Intrinsics::intrinsic_str_member_functions.begin(); it != Intrinsics::intrinsic_str_member_functions.end(); ++it)
117 possible.push_back(it->first);
118 } break;
119 case earl::value::Type::Char: {
120 for (auto it = Intrinsics::intrinsic_char_member_functions.begin(); it != Intrinsics::intrinsic_char_member_functions.end(); ++it)
121 possible.push_back(it->first);
122 } break;
123 case earl::value::Type::Option: {
124 for (auto it = Intrinsics::intrinsic_option_member_functions.begin(); it != Intrinsics::intrinsic_option_member_functions.end(); ++it)
125 possible.push_back(it->first);
126 } break;
127 case earl::value::Type::File: {
128 for (auto it = Intrinsics::intrinsic_file_member_functions.begin(); it != Intrinsics::intrinsic_file_member_functions.end(); ++it)
129 possible.push_back(it->first);
130 } break;
131 case earl::value::Type::Tuple: {
132 for (auto it = Intrinsics::intrinsic_tuple_member_functions.begin(); it != Intrinsics::intrinsic_tuple_member_functions.end(); ++it)
133 possible.push_back(it->first);
134 } break;
135 case earl::value::Type::DictInt:
136 case earl::value::Type::DictStr:
137 case earl::value::Type::DictFloat:
138 case earl::value::Type::DictChar: {
139 for (auto it = Intrinsics::intrinsic_dict_member_functions.begin(); it != Intrinsics::intrinsic_dict_member_functions.end(); ++it)
140 possible.push_back(it->first);
141 } break;
142 default: {
143 return identifier_not_declared(given, possible);
144 } break;
145 }
146
147 return identifier_not_declared(given, possible, /*include_intrinsics=*/false);
148}

150static std::shared_ptr<earl::value::Obj>
151eval_stmt_let_wmultiple_vars_wcustom_buffer_in_class(StmtLet *stmt,
152 std::unordered_map<std::string, std::shared_ptr<earl::variable::Obj>> &buffer,
153 std::shared_ptr<Ctx> &ctx,
154 bool ref) {
155 bool _ref = (stmt->m_attrs & static_cast<uint32_t>(Attr::Ref)) != 0;
156 bool _const = (stmt->m_attrs & static_cast<uint32_t>(Attr::Const)) != 0;
157 std::shared_ptr<earl::value::Obj> value = nullptr;
158 ER rhs = Interpreter::eval_expr(stmt->m_expr.get(), ctx, _ref);
159
160 {
161 int i = 0;
162 for (auto &id : stmt->m_ids) {
163 if (dynamic_cast<ClassCtx *>(ctx.get())->variable_exists_wo__m_class_constructor_tmp_args(id->lexeme())) {
164 std::string msg = "variable `"+id->lexeme()+"` is already declared";
165 auto conflict = ctx->variable_get(id->lexeme());
166 Err::err_wconflict(stmt->m_ids.at(i).get(), conflict->gettok());
167 throw InterpreterException(msg);
168 ++i;
169 }
170 }
171 }
172
173 if (rhs.is_ident() && buffer.find(rhs.id) != buffer.end())
174 value = buffer.find(rhs.id)->second->value();
175 else
176 value = unpack_ER(rhs, ctx, ref);
177
178 if (!rhs.is_class_instant()) {
179 PackedERPreliminary perp(nullptr);
180 value = unpack_ER(rhs, ctx, _ref, /*perp=*/&perp);
181 }
182 else
183 value = unpack_ER(rhs, ctx, _ref);
184
185 if (value->type() != earl::value::Type::Tuple) {
186 Err::err_wexpr(stmt->m_expr.get());
187 const std::string msg = "cannot declare multiple variables that do not equate to a tuple expression";
188 throw InterpreterException(msg);
189 }
190
191 auto tuple = dynamic_cast<earl::value::Tuple *>(value.get());
192
193 if (tuple->value().size() != stmt->m_ids.size()) {
194 Err::err_wexpr(stmt->m_expr.get());
195 const std::string msg =
196 "the number of variables declared ("
197 +std::to_string(stmt->m_ids.size())
198 +") does not match the size of the tuple expression ("
199 +std::to_string(tuple->value().size())
200 +")";
201 throw InterpreterException(msg);
202 }
203
204 int i = 0;
205 for (auto &tok : stmt->m_ids) {
206 if (stmt->m_tys.size() != 0)
207 typecheck(stmt->m_tys.at(i).get(), tuple->value().at(i).get(), ctx);
208
209 if (tok->lexeme() != "_") {
210 if (_const)
211 tuple->value().at(i)->set_const();
212
213 std::shared_ptr<earl::variable::Obj> var
214 = std::make_shared<earl::variable::Obj>(stmt->m_ids.at(i).get(), tuple->value().at(i), stmt->m_attrs);
215 ctx->variable_add(var);
216 }
217 ++i;
218 }
219
220 return std::make_shared<earl::value::Void>();
221}

223static std::shared_ptr<earl::value::Obj>
224eval_stmt_let_wcustom_buffer_in_class(StmtLet *stmt,
225 std::unordered_map<std::string, std::shared_ptr<earl::variable::Obj>> &buffer,
226 std::shared_ptr<Ctx> &ctx,
227 bool ref) {
228 if (stmt->m_ids.size() > 1)
229 return eval_stmt_let_wmultiple_vars_wcustom_buffer_in_class(stmt, buffer, ctx, ref);
230
231 assert(ctx->type() == CtxType::Class);
232
233 const std::string &id = stmt->m_ids.at(0)->lexeme();
234 if (dynamic_cast<ClassCtx *>(ctx.get())->variable_exists_wo__m_class_constructor_tmp_args(id)) {
235 std::string msg = "variable `"+id+"` is already declared";
236 auto conflict = ctx->variable_get(id);
237 Err::err_wconflict(stmt->m_ids.at(0).get(), conflict->gettok());
238 throw InterpreterException(msg);
239 }
240
241 bool _ref = (stmt->m_attrs & static_cast<uint32_t>(Attr::Ref)) != 0;
242 std::shared_ptr<earl::value::Obj> value = nullptr;
243 ER rhs = Interpreter::eval_expr(stmt->m_expr.get(), ctx, _ref);
244
245 if (rhs.is_ident() && buffer.find(rhs.id) != buffer.end())
246 value = buffer.find(rhs.id)->second->value();
247 else
248 value = unpack_ER(rhs, ctx, ref);
249
250 if (!rhs.is_class_instant()) {
251 PackedERPreliminary perp(nullptr);
252 value = unpack_ER(rhs, ctx, _ref, /*perp=*/&perp);
253 }
254 else
255 value = unpack_ER(rhs, ctx, _ref);
256
257 if (stmt->m_tys.size() > 0)
258 typecheck(stmt->m_tys[0].get(), value.get(), ctx);
259
260 if (id == "_")
261 return std::make_shared<earl::value::Void>();
262
263 auto info = flatten_info(stmt->m_info);
264
265 std::shared_ptr<earl::variable::Obj> var
266 = std::make_shared<earl::variable::Obj>(stmt->m_ids.at(0).get(), value, stmt->m_attrs, std::move(info));
267 ctx->variable_add(var);
268 value->set_owner(var.get());
269 return std::make_shared<earl::value::Void>();
270}

272static std::shared_ptr<earl::value::Obj>
273eval_class_instantiation(ExprFuncCall *expr,
274 const std::string &id,
275 std::vector<std::shared_ptr<earl::value::Obj>> &params,
276 std::shared_ptr<Ctx> &ctx,
277 bool ref) {
278 (void)ref;
279 StmtClass *class_stmt = nullptr;
280
281 if (ctx->type() == CtxType::Class) {
282 auto owner_ctx = dynamic_cast<WorldCtx *>(dynamic_cast<ClassCtx *>(ctx.get())->get_owner().get());
283 assert(owner_ctx);
284 class_stmt = owner_ctx->class_get(id);
285 }
286 else if (ctx->type() == CtxType::Function) {
287 auto functx = dynamic_cast<FunctionCtx *>(ctx.get());
288 if (functx->get_owner()->type() == CtxType::World)
289 class_stmt = dynamic_cast<WorldCtx *>(functx->get_owner().get())->class_get(id);
290 else {
291 auto class_ctx = functx->get_outer_class_owner_ctx();
292
293 assert(class_ctx);
294 assert(class_ctx->type() == CtxType::Class);
295
296 auto owner = dynamic_cast<ClassCtx*>(class_ctx.get())->get_owner();
297 auto world_ctx = dynamic_cast<WorldCtx *>(dynamic_cast<ClassCtx *>(class_ctx.get())->get_world_owner().get());
298
299 assert(world_ctx);
300 assert(world_ctx->type() == CtxType::World);
301
302 class_stmt = world_ctx->class_get(id);
303 }
304 }
305 else
306 class_stmt = dynamic_cast<WorldCtx *>(ctx.get())->class_get(id);
307
308 auto class_ctx = std::make_shared<ClassCtx>(ctx);
309
310 if (params.size() != class_stmt->m_constructor_args.size()) {
311 std::string msg = "Class `" + id + "` expects " + std::to_string(class_stmt->m_constructor_args.size()) +
312 " arguments but " + std::to_string(params.size()) + " were supplied";
313 Err::err_wexpr(expr);
314 throw InterpreterException(msg);
315 }
316
317 auto klass = std::make_shared<earl::value::Class>(class_stmt, class_ctx);
318
319 if (klass->is_experimental()) {
320 std::cerr << "warning: class `"+klass->id()+"` is marked as `experimental`" << std::endl;
321 klass->disable_experimental_flag();
322 }
323
324 // Add the constructor arguments to a temporary pushed scope
325 for (size_t i = 0; i < class_stmt->m_constructor_args.size(); ++i) {
326 __Type *ty = nullptr;
327
328 if (class_stmt->m_constructor_args[i].second.has_value())
329 ty = class_stmt->m_constructor_args[i].second.value().get();
330
331 if (ty) {
332 Interpreter::typecheck(ty, params[i].get(), ctx);
333 }
334
335 auto var = std::make_shared<earl::variable::Obj>(class_stmt->m_constructor_args[i].first.get(), params[i]);
336
337 // MAKE SURE TO CLEAR AT THE END OF THIS FUNC!
338 class_ctx->fill___m_class_constructor_tmp_args(var);
339 }
340
341 // Eval member variables
342 for (auto &member : class_stmt->m_members)
343 (void)eval_stmt_let_wcustom_buffer_in_class(member.get(),
344 class_ctx->get___m_class_constructor_tmp_args(),
345 klass->ctx(),
346 true);
347
348 const std::string constructor_id = "constructor";
349 bool has_constructor = false;
350
351 // Eval methods
352 for (size_t i = 0; i < class_stmt->m_methods.size(); ++i) {
353 (void)eval_stmt_def(class_stmt->m_methods[i].get(), klass->ctx());
354 if (class_stmt->m_methods[i]->m_id->lexeme() == constructor_id)
355 has_constructor = true;
356 }
357
358 if (has_constructor) {
359 std::vector<std::shared_ptr<earl::value::Obj>> unused = {};
360 (void)eval_user_defined_function(nullptr, constructor_id, unused, klass->ctx());
361 }
362
363 // CLEARED!
364 class_ctx->clear___m_class_constructor_tmp_args();
365
366 return klass;
367}

369static std::vector<std::shared_ptr<earl::value::Obj>>
370evaluate_function_parameters(ExprFuncCall *funccall, std::shared_ptr<Ctx> ctx, bool ref) {
371 std::vector<std::shared_ptr<earl::value::Obj>> res = {};
372 PackedERPreliminary perp(nullptr, /*this_=*/false, /*errtok=*/funccall->m_tok.get());
373 for (size_t i = 0; i < funccall->m_params.size(); ++i) {
374 ER er = Interpreter::eval_expr(funccall->m_params[i].get(), ctx, ref);
375 res.push_back(unpack_ER(er, ctx, ref, /*perp=*/&perp));
376 }
377 return res;
378}

380static std::vector<std::shared_ptr<earl::value::Obj>>
381evaluate_function_parameters_wrefs(ExprFuncCall *funccall,
382 std::variant<std::shared_ptr<earl::function::Obj>, earl::value::Closure *> &func_proper,
383 std::shared_ptr<Ctx> ctx) {
384 std::vector<std::shared_ptr<earl::value::Obj>> res = {};
385 std::vector<int> refs = {};
386
387 std::visit([&](auto &&fun) {
388 using T = std::decay_t<decltype(fun)>;
389 if constexpr (std::is_same_v<T, std::shared_ptr<earl::function::Obj>>) {
390
391 // Build reference table
392 for (size_t i = 0; i < fun->params_len(); ++i)
393 refs.push_back(static_cast<bool>(fun->param_at_is_ref(i)));
394
395 if (refs.size() != funccall->m_params.size()) {
396 const std::string msg = "function `"+fun->id()+"` expects "+std::to_string(refs.size())+" arguments but got "+std::to_string(funccall->m_params.size());
397 Err::err_wexpr(funccall);
398 throw InterpreterException(msg);
399 }
400
401 // Evaluate the parameters based on the reference table
402 for (size_t i = 0; i < funccall->m_params.size(); ++i) {
403 ER er = Interpreter::eval_expr(funccall->m_params[i].get(), ctx, /*ref=*/refs.at(i));
404 res.push_back(unpack_ER(er, ctx, refs[i]));
405 }
406 }
407
408 else if constexpr (std::is_same_v<T, earl::value::Closure *>) {
409
410 // Build reference table
411 for (size_t i = 0; i < fun->params_len(); ++i)
412 refs.push_back(static_cast<bool>(fun->param_at_is_ref(i)));
413
414 if (refs.size() != funccall->m_params.size()) {
415 const std::string msg = "closure `"+fun->tok()->lexeme()+"` expects "+std::to_string(refs.size())+" arguments but got "+std::to_string(funccall->m_params.size());
416 Err::err_wexpr(funccall);
417 throw InterpreterException(msg);
418 }
419
420 // Evaluate the parameters based on the reference table
421 for (size_t i = 0; i < funccall->m_params.size(); ++i) {
422 ER er = Interpreter::eval_expr(funccall->m_params[i].get(), ctx, /*ref=*/refs[i]);
423 res.push_back(unpack_ER(er, ctx, refs[i]));
424 }
425 }
426
427 else {
428 std::string msg = "A serious internal error has ocured and has gotten to an unreachable case. Something is very wrong";
429 throw InterpreterException(msg);
430 }
431 }, func_proper);
432
433 return res;
434}

436static std::shared_ptr<earl::value::Obj>
437eval_user_defined_function_wo_params(const std::string &id,
438 ExprFuncCall *funccall,
439 std::shared_ptr<Ctx> &funccall_ctx,
440 std::shared_ptr<Ctx> &ctx,
441 bool from_outside = false) {
442 std::vector<std::shared_ptr<earl::value::Obj>> params = {};
443 std::vector<bool> originally_was_const = {};
444 std::vector<int> refs = {};
445 std::variant<std::shared_ptr<earl::function::Obj>, earl::value::Closure *> v;
446 std::shared_ptr<earl::variable::Obj> maybe_closure = nullptr;
447
448 bool func_exists = false, var_exists = false;
449 func_exists = ctx->function_exists(id);
450 if (!func_exists) {
451 var_exists = ctx->variable_exists(id);
452 if (var_exists) {
453 maybe_closure = ctx->variable_get(id);
454 var_exists = maybe_closure->type() != earl::value::Type::Closure;
455 }
456 }
457
458 if (func_exists || var_exists) {
459 if (((flags & __SHOWFUNS) != 0) || ((flags & __VERBOSE) != 0))
460 std::cout << "[EARL show-funs] " << id << std::endl;
461
462 std::shared_ptr<earl::function::Obj> func = nullptr;
463
464 if (func_exists)
465 func = ctx->function_get(id);
466 else {
467 // std::shared_ptr<earl::variable::Obj> var = ctx->variable_get(id);
468 if (maybe_closure->type() == earl::value::Type::Closure)
469 goto is_closure;
470 if (maybe_closure->type() != earl::value::Type::FunctionRef)
471 goto bad;
472 earl::value::FunctionRef *ref = dynamic_cast<earl::value::FunctionRef *>(maybe_closure->value().get());
473 func = std::dynamic_pointer_cast<earl::function::Obj>(ref->value());
474 }
475
476 v = func;
477 if (from_outside && !func->is_pub()) {
478 std::string msg = "function `" + id + "` does not contain the @pub attribute";
479 Err::err_wexpr(funccall);
480 throw InterpreterException(msg);
481 }
482
483 params = evaluate_function_parameters_wrefs(funccall, v, funccall_ctx);
484 for (auto &p : params) {
485 if (p->is_const())
486 originally_was_const.push_back(true);
487 else
488 originally_was_const.push_back(false);
489 }
490
491 if (func->params_len() != params.size()) {
492 const std::string msg = "function `"+func->id()+"` expects "+std::to_string(func->params_len())+" arguments but got "+std::to_string(params.size());
493 Err::err_wexpr(funccall);
494 throw InterpreterException(msg);
495 }
496
497 auto fctx = std::make_shared<FunctionCtx>(ctx, func->attrs());
498 fctx->set_curfunc(id);
499 func->load_parameters(params, fctx, ctx);
500
501 // Recursion optimization
502 if (ctx->type() == CtxType::Function) {
503 if (fctx->get_curfuncid() == dynamic_cast<FunctionCtx *>(ctx.get())->get_curfuncid()) {
504 fctx->setrec();
505 }
506 }
507
508 if ((func->attrs() & static_cast<uint32_t>(Attr::Experimental)) != 0) {
509 std::cerr << "warning: function `"+func->id()+"` is marked as `experimental`" << std::endl;
510 func->disable_experimental_flag();
511 }
512
513 std::shared_ptr<Ctx> mask = fctx;
514 auto res = Interpreter::eval_stmt_block(func->block(), mask);
515
516 for (size_t i = 0; i < originally_was_const.size(); ++i) {
517 if (!originally_was_const[i])
518 params[i]->unset_const();
519 }
520
521 if (func->is_explicit_typed()) {
522 auto ty = func->get_explicit_type();
523 Interpreter::typecheck(ty, res.get(), ctx);
524 }
525
526 return res;
527 }
528 else if (ctx->closure_exists(id)) {
529 is_closure:
530 auto cl = maybe_closure ? std::move(maybe_closure) : ctx->variable_get(id);
531 auto clctx = std::make_shared<ClosureCtx>(ctx);
532 earl::value::Closure *clvalue = dynamic_cast<earl::value::Closure *>(cl->value().get());
533 v = clvalue;
534 params = evaluate_function_parameters_wrefs(funccall, v, funccall_ctx);
535 if (clvalue->params_len() != params.size()) {
536 const std::string msg = "closure `"+id+"` expects "+std::to_string(clvalue->params_len())+" arguments but got "+std::to_string(params.size());
537 Err::err_wexpr(funccall);
538 throw InterpreterException(msg);
539 }
540 clvalue->load_parameters(params, clctx);
541 std::shared_ptr<Ctx> mask = clctx;
542 return Interpreter::eval_stmt_block(clvalue->block(), mask);
543 }
544
545bad:
546 Err::err_wexpr(funccall);
547
548 std::string msg = "function `" + id + "` has not been defined\n";
549 auto avail = ctx->get_available_function_names();
550 msg += "did you mean: " + identifier_not_declared(id, avail) + "?";
551
552 throw InterpreterException(msg);
553 return nullptr; // unreachable
554}

556static std::shared_ptr<earl::value::Obj>
557eval_user_defined_function(ExprFuncCall *expr,
558 const std::string &id,
559 std::vector<std::shared_ptr<earl::value::Obj>> &params,
560 std::shared_ptr<Ctx> &ctx,
561 bool from_outside) {
562 bool func_exists = false, var_exists = false;
563 func_exists = ctx->function_exists(id);
564 std::shared_ptr<earl::variable::Obj> maybe_closure = nullptr;
565 if (!func_exists) {
566 var_exists = ctx->variable_exists(id);
567 if (var_exists) {
568 maybe_closure = ctx->variable_get(id);
569 var_exists = maybe_closure->type() != earl::value::Type::Closure;
570 }
571 }
572
573 if (func_exists || var_exists) {
574 if ((flags & __SHOWFUNS) != 0)
575 std::cout << "[EARL show-funs] " << id << '\n';
576
577 std::shared_ptr<earl::function::Obj> func = nullptr;
578
579 if (func_exists)
580 func = ctx->function_get(id);
581 else {
582 // std::shared_ptr<earl::variable::Obj> var = ctx->variable_get(id);
583 if (maybe_closure->type() == earl::value::Type::Closure)
584 goto is_closure;
585 if (maybe_closure->type() != earl::value::Type::FunctionRef)
586 goto bad;
587 earl::value::FunctionRef *ref = dynamic_cast<earl::value::FunctionRef *>(maybe_closure->value().get());
588 func = std::dynamic_pointer_cast<earl::function::Obj>(ref->value());
589 }
590
591 if (func->params_len() != params.size()) {
592 const std::string msg = "function `"+func->id()+"` expects "+std::to_string(func->params_len())+" arguments but got "+std::to_string(params.size());
593 Err::err_wexpr(expr);
594 throw InterpreterException(msg);
595 }
596 if (from_outside && !func->is_pub()) {
597 std::string msg = "function `"+id+"` does not contain the @pub attribute";
598 if (expr)
599 Err::err_wexpr(expr);
600 throw InterpreterException(msg);
601 }
602 auto fctx = std::make_shared<FunctionCtx>(ctx, func->attrs());
603 func->load_parameters(params, fctx, ctx);
604 fctx->set_curfunc(id);
605 func->load_parameters(params, fctx, ctx);
606
607 if (ctx->type() == CtxType::Function) {
608 if (fctx->get_curfuncid() == dynamic_cast<FunctionCtx *>(ctx.get())->get_curfuncid()) {
609 fctx->setrec();
610 }
611 }
612
613 std::shared_ptr<Ctx> mask = fctx;
614
615 if ((func->attrs() & static_cast<uint32_t>(Attr::Experimental)) != 0)
616 std::cerr << "warning: function `"+func->id()+"` is marked as `experimental`" << std::endl;;
617
618 auto res = Interpreter::eval_stmt_block(func->block(), mask);
619
620 if (func->is_explicit_typed()) {
621 auto ty = func->get_explicit_type();
622 Interpreter::typecheck(ty, res.get(), ctx);
623 }
624
625 return res;
626 }
627 else if (ctx->closure_exists(id)) {
628 is_closure:
629 // auto cl = ctx->variable_get(id);
630 auto cl = maybe_closure ? std::move(maybe_closure) : ctx->variable_get(id);
631 auto clctx = std::make_shared<ClosureCtx>(ctx);
632 auto clvalue = dynamic_cast<earl::value::Closure *>(cl->value().get());
633 if (clvalue->params_len() != params.size()) {
634 const std::string msg = "closure `"+id+"` expects "+std::to_string(clvalue->params_len())+" arguments but got "+std::to_string(params.size());
635 Err::err_wexpr(expr);
636 throw InterpreterException(msg);
637 }
638 clvalue->load_parameters(params, clctx);
639 std::shared_ptr<Ctx> mask = clctx;
640 return Interpreter::eval_stmt_block(clvalue->block(), mask);
641 }
642
643 bad:
644
645 Err::err_wexpr(expr);
646
647 std::string msg = "function `" + id + "` has not been defined\n";
648 auto avail = ctx->get_available_function_names();
649 msg += "did you mean: " + identifier_not_declared(id, avail) + "?";
650
651 throw InterpreterException(msg);
652 return nullptr; // unreachable
653}

655static std::shared_ptr<earl::value::Obj>
656unpack_ER(ER &er, std::shared_ptr<Ctx> &ctx, bool ref, PackedERPreliminary *perp) {
657 if (er.value && er.value->type() == earl::value::Type::Return)
658 er.value = std::make_shared<earl::value::Void>();
659
660 // CLASSES
661 if (er.is_class_instant()) {
662 auto params = evaluate_function_parameters(static_cast<ExprFuncCall *>(er.extra), ctx, ref);
663 auto class_instantiation = eval_class_instantiation(static_cast<ExprFuncCall *>(er.extra), er.id, params, ctx, ref);
664 return class_instantiation;
665 }
666
667 // FUNCTIONS/MEMBERS/INTRINSICS
668 if (er.is_function_ident()) {
669 auto params = evaluate_function_parameters(static_cast<ExprFuncCall *>(er.extra), er.ctx, ref);
670 if (er.is_intrinsic()) {
671 Expr *expr = nullptr;
672 if (er.extra)
673 expr = static_cast<Expr *>(er.extra);
674 auto call = Intrinsics::call(er.id, params, ctx, expr);
675 if (call->type() == earl::value::Type::Return)
676 call = std::make_shared<earl::value::Void>();
677 return call;
678 }
679
680 if (er.is_member_intrinsic() && (perp && perp->lhs_getter_accessor)) {
681 if (Intrinsics::is_member_intrinsic(er.id, static_cast<int>(perp->lhs_getter_accessor->type()))) {
682 Expr *expr = nullptr;
683 if (er.extra) expr = static_cast<Expr *>(er.extra);
684 auto res = Intrinsics::call_member(er.id,
685 perp->lhs_getter_accessor->type(),
686 perp->lhs_getter_accessor,
687 params,
688 ctx,
689 expr);
690 return res;
691 }
692 }
693
694 if (ctx->type() == CtxType::Class) {
695 std::shared_ptr<earl::function::Obj> func = nullptr;
696 if (ctx->function_exists(er.id))
697 func = ctx->function_get(er.id);
698 if ((!perp || !perp->this_) && (func && er.ctx != ctx && !func->is_pub())) {
699 std::string msg = "member `" + func->id() + "` is missing the @pub attribute";
700 if (er.extra) Err::err_wexpr(static_cast<Expr *>(er.extra));
701 throw InterpreterException(msg);
702 }
703
704 auto call = eval_user_defined_function(static_cast<ExprFuncCall *>(er.extra), er.id, params, ctx);
705 if (call->type() == earl::value::Type::Return)
706 call = std::make_shared<earl::value::Void>();
707 return call;
708 }
709
710 // Method does not exist
711 if (perp && perp->lhs_getter_accessor) {
712 if (er.extra) Err::err_wexpr(static_cast<Expr *>(er.extra));
713 std::string msg = "method `" + er.id + "` is not a part of the given type `"+earl::value::type_to_str(perp->lhs_getter_accessor->type())+"`\n";
714 // auto avail = ctx->get_available_function_names();
715 msg += "did you mean: " + method_not_declared(er.id, perp->lhs_getter_accessor) + "?";
716 throw InterpreterException(msg);
717 }
718
719 // Check for class reference
720 if (ctx->variable_exists(er.id)) {
721 auto var = ctx->variable_get(er.id);
722 if (var->type() == earl::value::Type::ClassRef) {
723 auto value = dynamic_cast<earl::value::ClassRef *>(var->value().get());
724 auto class_instantiation = eval_class_instantiation(static_cast<ExprFuncCall *>(er.extra), value->get_stmt()->m_id->lexeme(), params, ctx, ref);
725 return class_instantiation;
726 }
727 }
728
729 // We need to have this function to gen the parameters so we
730 // know which ones need to be taken as a reference. NOTE: The
731 // routine(s) above this may need this change as well.
732 auto call = eval_user_defined_function_wo_params(er.id, static_cast<ExprFuncCall *>(er.extra), er.ctx, ctx);
733 if (call->type() == earl::value::Type::Return)
734 call = std::make_shared<earl::value::Void>();
735 return call;
736 }
737
738 // LITERAL
739 else if (er.is_literal())
740 return er.value;
741
742 // IDENTIFIER
743 else if (er.is_ident()) {
744 // Check if it is an entry of an enum
745 if (perp && perp->lhs_getter_accessor && perp->lhs_getter_accessor->type() == earl::value::Type::Enum) {
746 auto lhs = dynamic_cast<earl::value::Enum *>(perp->lhs_getter_accessor.get());
747 if (lhs->has_entry(er.id)) {
748 if (lhs->is_experimental()) {
749 std::cerr << "warning: enum `"+lhs->id()+"` is marked as `experimental`" << std::endl;
750 lhs->disable_experimental_flag();
751 }
752 return lhs->get_entry(er.id)->value()->copy();
753 }
754 }
755 if (ctx->variable_exists(er.id)) {
756 auto var = ctx->variable_get(er.id);
757 if ((!perp || !perp->this_) && (er.ctx != ctx && !var->is_pub())) {
758 std::string msg = "member variable `"+var->id()+"` is missing the @pub attribute";
759 if (er.extra) Err::err_wexpr(static_cast<Expr *>(er.extra));
760 throw InterpreterException(msg);
761 }
762 if (!ref)
763 return var->value()->copy();
764 return var->value();
765 }
766 // Check if it is an enum
767 WorldCtx *world = ctx->type() == CtxType::World ?
768 dynamic_cast<WorldCtx *>(ctx.get()) :
769 ctx->get_world();
770 if (world->enum_exists(er.id)) {
771 return world->enum_get(er.id);
772 }
773
774 // Check if it is a type as a value
775 if (earl::value::is_typekw(er.id)) {
776 return std::make_shared<earl::value::TypeKW>(earl::value::get_typekw_proper(er.id));
777 }
778
779 // Check for builtin identifier
780 if (earl::value::is_builtin_ident(er.id)) {
781 return earl::value::get_builtin_ident(er.id, ctx);
782 }
783
784 // Check for a function identifier (not the same as calling a function).
785 if (ctx->function_exists(er.id)) {
786 auto fun = ctx->function_get(er.id);
787 return std::make_shared<earl::value::FunctionRef>(fun);
788 }
789
790 // Check for a class identifier (not the same as calling a function).
791 if (world->class_is_defined(er.id)) {
792 auto klass = world->class_get(er.id);
793 return std::make_shared<earl::value::ClassRef>(klass);
794 }
795
796 Err::err_wexpr(static_cast<Expr *>(er.extra));
797
798 std::string msg = "variable `" + er.id + "` has not been defined\n";
799 auto avail = ctx->get_available_variable_names();
800
801 if (avail.size() != 0)
802 msg += "did you mean: " + identifier_not_declared(er.id, avail, /*include_intrinsics=*/false) + "?";
803
804 throw InterpreterException(msg);
805 }
806
807 // UNIT
808 else if (er.is_wildcard())
809 return std::make_shared<earl::value::Void>();
810 else
811 assert(false && "unreachable");
812 return nullptr; // unreachable
813}

838static ER
839eval_expr_term_funccall(ExprFuncCall *expr, std::shared_ptr<Ctx> &ctx, bool ref) {
840
841 // Checks if `_id` is a class in the @world scope.
842 std::function<std::shared_ptr<Ctx>(const std::string &, std::shared_ptr<Ctx> &)> check_if_is_class
843 = [&](const std::string &_id, std::shared_ptr<Ctx> &_ctx) -> std::shared_ptr<Ctx> {
844 if (_ctx->type() == CtxType::World && dynamic_cast<WorldCtx *>(_ctx.get())->class_is_defined(_id))
845 return _ctx;
846 if (_ctx->type() == CtxType::Class)
847 return check_if_is_class(_id, dynamic_cast<ClassCtx *>(_ctx.get())->get_owner());
848 if (_ctx->type() == CtxType::Function)
849 return check_if_is_class(_id, dynamic_cast<FunctionCtx *>(_ctx.get())->get_owner());
850 if (_ctx->type() == CtxType::Closure)
851 return check_if_is_class(_id, dynamic_cast<ClosureCtx *>(_ctx.get())->get_owner());
852 return nullptr;
853 };
854
855 ER left = Interpreter::eval_expr(expr->m_left.get(), ctx, ref);
856 const std::string &id = left.id;
857
858 if (Intrinsics::is_intrinsic(id))
859 return ER(nullptr, static_cast<ERT>(ERT::FunctionIdent|ERT::IntrinsicFunction), /*id=*/id, /*extra=*/static_cast<void *>(expr), /*ctx=*/ctx);
860
861 if (Intrinsics::is_member_intrinsic(id))
862 return ER(nullptr, static_cast<ERT>(ERT::FunctionIdent|ERT::IntrinsicMemberFunction), /*id=*/id, /*extra=*/static_cast<void *>(expr), /*ctx=*/ctx);
863
864 std::shared_ptr<Ctx> ctx_wclass = check_if_is_class(id, ctx);
865 if (ctx_wclass)
866 return ER(nullptr, static_cast<ERT>(ERT::ClassInstant|ERT::Literal), /*id=*/id, /*extra=*/static_cast<void *>(expr), /*ctx=*/ctx_wclass);
867
868 return ER(nullptr, ERT::FunctionIdent, /*id=*/id, /*extra=*/static_cast<void *>(expr), /*ctx=*/ctx);
869}

872ER
873eval_expr_term_mod_access(ExprModAccess *expr, std::shared_ptr<Ctx> &ctx, bool ref) {
874 ExprModAccess *mod_access = expr;
875 ExprIdent *left_ident = mod_access->m_expr_ident.get();
876 const auto &left_id = left_ident->m_tok->lexeme();
877 ER right_er(std::shared_ptr<earl::value::Obj>{}, ERT::None);
878
879 std::shared_ptr<Ctx> *ctx_ptr = nullptr;
880
881 if (ctx->type() == CtxType::World)
882 ctx_ptr = dynamic_cast<WorldCtx *>(ctx.get())->get_import(left_id);
883 else if (ctx->type() == CtxType::Function) {
884 auto world = dynamic_cast<FunctionCtx *>(ctx.get())->get_outer_world_owner();
885 ctx_ptr = dynamic_cast<WorldCtx *>(world.get())->get_import(left_id);
886 }
887 else if (ctx->type() == CtxType::Class) {
888 auto klass = dynamic_cast<ClassCtx *>(ctx.get());
889 auto world = dynamic_cast<WorldCtx *>(klass->get_world_owner().get());
890 ctx_ptr = world->get_import(left_id);
891 }
892 else if (ctx->type() == CtxType::Closure) {
893 auto world = dynamic_cast<ClosureCtx *>(ctx.get())->get_outer_world_owner();
894 ctx_ptr = dynamic_cast<WorldCtx *>(world.get())->get_import(left_id);
895 }
896
897 std::shared_ptr<Ctx> &other_ctx = *ctx_ptr;
898
899 std::visit([&](auto &&arg) {
900 using T = std::decay_t<decltype(arg)>;
901 if constexpr (std::is_same_v<T, std::unique_ptr<ExprIdent>>)
902 right_er = Interpreter::eval_expr(arg.get(), other_ctx, true);
903 else if constexpr (std::is_same_v<T, std::unique_ptr<ExprFuncCall>>)
904 right_er = Interpreter::eval_expr(arg.get(), other_ctx, true);
905 else {
906 std::string msg = "A serious internal error has ocured and has gotten to an unreachable case. Something is very wrong";
907 throw InterpreterException(msg);
908 }
909 }, expr->m_right);
910
911 if (right_er.is_class_instant()) {
912 assert(right_er.ctx->type() == CtxType::World);
913 WorldCtx *world = dynamic_cast<WorldCtx *>(right_er.ctx.get());
914 if (world->class_is_defined(right_er.id)) {
915 if ((world->class_get(right_er.id)->m_attrs & static_cast<uint32_t>(Attr::Pub)) == 0) {
916 Err::err_wexpr(expr);
917 std::string msg = "class `"+right_er.id+"` in module `"+left_id+"` does not contain the @pub attribute";
918 throw InterpreterException(msg);
919 }
920 }
921 auto params = evaluate_function_parameters(static_cast<ExprFuncCall *>(right_er.extra), ctx, ref);
922 auto class_instantiation = eval_class_instantiation(static_cast<ExprFuncCall *>(right_er.extra), right_er.id, params, right_er.ctx, ref);
923 return ER(class_instantiation, ERT::Literal, /*id=*/"", /*extra=*/nullptr, /*ctx=*/ctx);
924 }
925 if (right_er.is_function_ident()) {
926 // Check if the function has @pub attribute
927 if (right_er.ctx->function_exists(right_er.id)) {
928 if (!right_er.ctx->function_get(right_er.id)->is_pub()) {
929 std::string msg = "function `"+right_er.id+"` in module `"+left_id+"` does not contain the @pub attribute";
930 Err::err_wexpr(left_ident);
931 throw InterpreterException(msg);
932 }
933 }
934 // Check if the closure has @pub attribute
935 else if (right_er.ctx->closure_exists(right_er.id)) {
936 if (!right_er.ctx->function_get(right_er.id)->is_pub()) {
937 std::string msg = "variable `"+right_er.id+"` in module `"+left_id+"` does not contain the @pub attribute";
938 Err::err_wexpr(left_ident);
939 throw InterpreterException(msg);
940 }
941 }
942 auto func = eval_user_defined_function_wo_params(right_er.id, static_cast<ExprFuncCall *>(right_er.extra), ctx, right_er.ctx);
943 return ER(func, ERT::Literal);
944 }
945 else if (right_er.is_ident()) {
946 auto value = unpack_ER(right_er, other_ctx, ref);
947 // Check if it is a variable
948 if (right_er.ctx->variable_exists(right_er.id)) {
949 if (!right_er.ctx->variable_get(right_er.id)->is_pub()) {
950 std::string msg = "variable `"+right_er.id+"` in module `"+left_id+"` does not contain the @pub attribute";
951 Err::err_wexpr(left_ident);
952 throw InterpreterException(msg);
953 }
954 return ER(value, ERT::Literal);
955 }
956 // Check if it is a function identifier
957 if (right_er.ctx->function_exists(right_er.id)) {
958 if (!right_er.ctx->function_get(right_er.id)->is_pub()) {
959 std::string msg = "function `"+right_er.id+"` in module `"+left_id+"` does not contain the @pub attribute";
960 Err::err_wexpr(left_ident);
961 throw InterpreterException(msg);
962 }
963 return ER(value, ERT::Literal);
964 }
965
966 if (right_er.ctx->type() == CtxType::World) {
967 auto worldctx = dynamic_cast<WorldCtx *>(right_er.ctx.get());
968 if (worldctx->class_is_defined(right_er.id)) {
969 auto class_stmt = worldctx->class_get(right_er.id);
970 if ((class_stmt->m_attrs & static_cast<uint32_t>(Attr::Pub)) == 0) {
971 std::string msg = "class `"+right_er.id+"` in module `"+left_id+"` does not contain the @pub attribute";
972 Err::err_wexpr(left_ident);
973 throw InterpreterException(msg);
974 }
975 return ER(value, ERT::Literal);
976 }
977 }
978
979 // Check if it is a class identifier
980
981 // It must be an enum
982 assert(value->type() == earl::value::Type::Enum);
983 if (!dynamic_cast<earl::value::Enum *>(value.get())->is_pub()) {
984 std::string msg = "enumeration `"+right_er.id+"` in module `"+left_id+"` does not contain the @pub attribute";
985 Err::err_wexpr(left_ident);
986 throw InterpreterException(msg);
987 }
988 return ER(value, ERT::Literal);
989 }
990 else
991 assert(false && "unimplemented");
992}

994ER
995eval_expr_term_get(ExprGet *expr, std::shared_ptr<Ctx> &ctx, bool ref) {
996 ER left_er = Interpreter::eval_expr(expr->m_left.get(), ctx, ref);
997 ER right_er(std::shared_ptr<earl::value::Obj>{}, ERT::None);
998
999 std::visit([&](auto &&arg) {
1000 using T = std::decay_t<decltype(arg)>;
1001 if constexpr (std::is_same_v<T, std::unique_ptr<ExprIdent>>)
1002 right_er = Interpreter::eval_expr(arg.get(), ctx, true);
1003 else if constexpr (std::is_same_v<T, std::unique_ptr<ExprFuncCall>>)
1004 right_er = Interpreter::eval_expr(arg.get(), ctx, true);
1005 else {
1006 std::string msg = "A serious internal error has ocured and has gotten to an unreachable case. Something is very wrong";
1007 throw InterpreterException(msg);
1008 }
1009 }, expr->m_right);
1010
1011 if (left_er.id == "this") {
1012 if (ctx->type() != CtxType::Function && ctx->type() != CtxType::Closure) {
1013 std::string msg = "Must be in a function in a class context to use the `this` keyword";
1014 Err::err_wexpr(expr);
1015 throw InterpreterException(msg);
1016 }
1017
1018 std::shared_ptr<earl::value::Obj> value = nullptr;
1019
1020 if (ctx->type() == CtxType::Closure) {
1021 auto closure_ctx = dynamic_cast<ClosureCtx *>(ctx.get());
1022 if (!closure_ctx->in_class()) {
1023 std::string msg = "Must be in a class context when using the `this` keyword";
1024 Err::err_wexpr(expr);
1025 throw InterpreterException(msg);
1026 }
1027 PackedERPreliminary perp(nullptr, true);
1028 value = unpack_ER(right_er, closure_ctx->get_outer_class_owner_ctx(), /*ref=*/true, /*perp=*/&perp);
1029 }
1030 else if (ctx->type() == CtxType::Class) {
1031 PackedERPreliminary perp(nullptr, true);
1032 value = unpack_ER(right_er, ctx, /*ref=*/true, /*perp=*/&perp);
1033 }
1034 else if (ctx->type() == CtxType::Function) {
1035 auto fctx = dynamic_cast<FunctionCtx *>(ctx.get());
1036
1037 if (!fctx->in_class()) {
1038 std::string msg = "Must be in a class context when using the `this` keyword";
1039 Err::err_wexpr(expr);
1040 throw InterpreterException(msg);
1041 }
1042
1043 PackedERPreliminary perp(nullptr, true);
1044 value = unpack_ER(right_er, fctx->get_outer_class_owner_ctx(), /*ref=*/true, /*perp=*/&perp);
1045 }
1046 else {
1047 std::string msg = "Must be in a function in a class context to use the `this` keyword";
1048 Err::err_wexpr(expr);
1049 throw InterpreterException(msg);
1050 }
1051 return ER(value, ERT::Literal);
1052 }
1053 else {
1054 auto left_value = unpack_ER(left_er, ctx, true);
1055 PackedERPreliminary perp(left_value, /*this=*/false, /*errtok=*/expr->m_tok.get());
1056 std::shared_ptr<earl::value::Obj> value = nullptr;
1057
1058 if (left_value->type() == earl::value::Type::Class) {
1059 // Class method/member. The right side (right_er) contains the actual call/identifier to be evaluated,
1060 // and we need the left (left_value)'s context with the preliminary value of (perp).
1061 // auto cctx = dynamic_cast<earl::value::Class *>(left_value.get())->ctx();
1062 // dynamic_cast<ClassCtx *>(cctx.get())->function_debug_dump();
1063 value = unpack_ER(right_er, dynamic_cast<earl::value::Class *>(left_value.get())->ctx(), ref, &perp);
1064 }
1065 else
1066 // Function chaining and member intrinsics...
1067 value = unpack_ER(right_er, ctx, ref, &perp);
1068
1069 return ER(value, ERT::Literal);
1070 }
1071
1072 assert(false && "unreachable");
1073}

1093static ER
1094eval_expr_term_listlit(ExprListLit *expr, std::shared_ptr<Ctx> &ctx, bool ref) {
1095 std::vector<std::shared_ptr<earl::value::Obj>> list = {};
1096 for (size_t i = 0; i < expr->m_elems.size(); ++i) {
1097 ER er = Interpreter::eval_expr(expr->m_elems.at(i).get(), ctx, ref);
1098 list.push_back(unpack_ER(er, ctx, ref));

1181 ER right_er = Interpreter::eval_expr(expr->m_end.get(), ctx, ref);

1200 values.push_back(std::make_shared<earl::value::Int>(start++));

1216 while (start < end)

1218 }

1220 }

1222 std::string msg = "invalid type "+earl::value::type_to_str(lvalue->type())+"` for type range";
1223 Err::err_wexpr(expr->m_start.get());

1398static ER
1399eval_expr_term_fstr(ExprFStr *expr, std::shared_ptr<Ctx> &ctx, bool ref) {
1400 const std::string &str = expr->m_tok->lexeme();
1401 auto result = std::make_shared<earl::value::Str>();
1402
1403 auto until_closing = [](const std::string &s, size_t &it) -> std::string {
1404 std::string buf = "";
1405 while (it < s.size() && s[it] != '}') {
1406 buf.push_back(s.at(it));
1407 ++it;
1408 }
1409 return buf;
1410 };
1411
1412 for (size_t i = 0; i < str.size(); ++i) {
1413 char c = str.at(i);
1414 if (c == '{') {
1415 ++i;
1416 std::string id = until_closing(str, i);
1417 if (!ctx->variable_exists(id)) {
1418 Err::err_wexpr(expr);
1419
1420 std::string msg = "variable `" + id + "` has not been defined\n";
1421 auto avail = ctx->get_available_variable_names();
1422 if (avail.size() != 0)
1423 msg += "did you mean: " + identifier_not_declared(id, avail) + "?";
1424
1425 throw InterpreterException(msg);
1426 }
1427 auto var = ctx->variable_get(id);
1428 std::string strified = var->value()->to_cxxstring();
1429 result->append(strified);
1430 }
1431 else
1432 result->append(c);
1433 }
1434
1435 return ER(result, ERT::Literal);
1436}

1468ER
1469eval_expr_bin(ExprBinary *expr, std::shared_ptr<Ctx> &ctx, bool ref) {
1470 ER lhs = Interpreter::eval_expr(expr->m_lhs.get(), ctx, ref);
1471 auto lhs_value = unpack_ER(lhs, ctx, true);
1472
1473 // Short-circuit evaluation for logical AND (&&)
1474 if (expr->m_op->type() == TokenType::Double_Ampersand) {
1475 // If lhs is false, return lhs (no need to evaluate rhs)
1476 if (!lhs_value->boolean())
1477 return lhs;
1478 ER rhs = Interpreter::eval_expr(expr->m_rhs.get(), ctx, ref);
1479 return ER(unpack_ER(rhs, ctx, ref), ERT::Literal);
1480 }
1481
1482 // Short-circuit evaluation for logical OR (||)
1483 if (expr->m_op->type() == TokenType::Double_Pipe) {
1484 // If lhs is true, return lhs (no need to evaluate rhs)
1485 if (lhs_value->boolean())
1486 return lhs;
1487 ER rhs = Interpreter::eval_expr(expr->m_rhs.get(), ctx, ref);
1488 return ER(unpack_ER(rhs, ctx, ref), ERT::Literal);
1489 }
1490
1491 ER rhs = Interpreter::eval_expr(expr->m_rhs.get(), ctx, ref);
1492 auto rhs_value = unpack_ER(rhs, ctx, ref);
1493 // auto result = lhs_value->binop(expr->m_op.get(), rhs_value);
1494 std::shared_ptr<earl::value::Obj> result = nullptr;
1495 switch (expr->m_op.get()->type()) {
1496 case TokenType::Plus: {
1497 result = lhs_value->add(expr->m_op.get(), rhs_value.get());
1498 } break;
1499 case TokenType::Minus: {
1500 result = lhs_value->sub(expr->m_op.get(), rhs_value.get());
1501 } break;
1502 case TokenType::Asterisk: {
1503 result = lhs_value->multiply(expr->m_op.get(), rhs_value.get());
1504 } break;
1505 case TokenType::Forwardslash: {
1506 result = lhs_value->divide(expr->m_op.get(), rhs_value.get());
1507 } break;
1508 case TokenType::Percent: {
1509 result = lhs_value->modulo(expr->m_op.get(), rhs_value.get());
1510 } break;
1511 case TokenType::Double_Asterisk: {
1512 result = lhs_value->power(expr->m_op.get(), rhs_value.get());
1513 } break;
1514 case TokenType::Greaterthan:
1515 case TokenType::Lessthan:
1516 case TokenType::Greaterthan_Equals:
1517 case TokenType::Lessthan_Equals: {
1518 result = lhs_value->gtequality(expr->m_op.get(), rhs_value.get());
1519 } break;
1520 case TokenType::Double_Equals:
1521 case TokenType::Bang_Equals: {
1522 result = lhs_value->equality(expr->m_op.get(), rhs_value.get());
1523 } break;
1524 case TokenType::Backtick_Pipe:
1525 case TokenType::Backtick_Caret:
1526 case TokenType::Backtick_Ampersand: {
1527 result = lhs_value->bitwise(expr->m_op.get(), rhs_value.get());
1528 } break;
1529 case TokenType::Double_Lessthan:
1530 case TokenType::Double_Greaterthan: {
1531 result = lhs_value->bitshift(expr->m_op.get(), rhs_value.get());
1532 } break;
1533 default: {
1534 Err::err_wtok(expr->m_op.get());
1535 const std::string msg = "invalid operator";
1536 throw InterpreterException(msg);
1537 } break;
1538 }
1539 return ER(result, ERT::Literal);
1540}

1605 assert(false);
1606 }
1607
1608 const std::string &tyname = ty->m_main_ty->lexeme();
1609
1610 if (tyname == COMMON_EARLTY_ANY) return;
1611 else if (tyname == COMMON_EARLTY_INT32 && value->type() == earl::value::Type::Int) return;
1612 else if (tyname == COMMON_EARLTY_FLOAT && value->type() == earl::value::Type::Float) return;
1613 else if (tyname == COMMON_EARLTY_STR && value->type() == earl::value::Type::Str) return;
1614 else if (tyname == COMMON_EARLTY_UNIT
1615 && (value->type() == earl::value::Type::Void
1616 || value->type() == earl::value::Type::Return)) return;
1617 else if (tyname == COMMON_EARLTY_CHAR && value->type() == earl::value::Type::Char) return;
1618 else if (tyname == COMMON_EARLTY_BOOL && value->type() == earl::value::Type::Bool) return;
1619 else if (tyname == COMMON_EARLTY_LIST && value->type() == earl::value::Type::List) return;
1620 else if (tyname == COMMON_EARLTY_TUPLE && value->type() == earl::value::Type::Tuple) return;
1621 else if (tyname == COMMON_EARLTY_FILE && value->type() == earl::value::Type::File) return;
1622 else if (tyname == COMMON_EARLTY_CLOSURE && value->type() == earl::value::Type::Closure) return;
1623 else if (tyname == COMMON_EARLTY_OPTION && value->type() == earl::value::Type::Option) return;
1624 else if (tyname == COMMON_EARLTY_SLICE && value->type() == earl::value::Type::Slice) return;
1625 else if (tyname == COMMON_EARLTY_DICT
1626 && (value->type() == earl::value::Type::DictInt
1627 || value->type() == earl::value::Type::DictStr
1628 || value->type() == earl::value::Type::DictFloat
1629 || value->type() == earl::value::Type::DictChar)) return;
1630 else if (tyname == COMMON_EARLTY_TYPE && value->type() == earl::value::Type::TypeKW) return;
1631 else if (tyname == COMMON_EARLTY_REAL
1632 && (value->type() == earl::value::Type::Int
1633 || value->type() == earl::value::Type::Float)) return;
1634 else if (value->type() == earl::value::Type::Class) {
1635 auto klass = dynamic_cast<earl::value::Class *>(value);
1636 if (klass->id() == tyname)
1637 return;
1638 }
1639
1640 Err::err_wtok(ty->m_main_ty.get());
1641 std::string msg = "";
1642 if (value->type() == earl::value::Type::Class)
1643 msg = "explicit type of `"+tyname+"` does not match what was given `"+dynamic_cast<earl::value::Class *>(value)->id()+"`";
1644 else
1645 msg = "explicit type of `"+tyname+"` does not match what was given `"+type_to_str(value->type())+"`";
1646 throw InterpreterException(msg);
1647}

1899std::shared_ptr<earl::value::Obj>
1900eval_stmt_mut(StmtMut *stmt, std::shared_ptr<Ctx> &ctx) {
1901 ER left_er = Interpreter::eval_expr(stmt->m_left.get(), ctx, true);
1902 ER right_er = Interpreter::eval_expr(stmt->m_right.get(), ctx, false);
1903
1904 if (left_er.is_tuple_access()) {
1905 Err::err_wexpr(stmt->m_left.get());
1906 Err::err_wexpr(stmt->m_right.get());
1907 std::string msg = "cannot mutate tuple type as they are immutable";
1908 throw InterpreterException(msg);
1909 }
1910
1911 auto l = unpack_ER(left_er, ctx, true);
1912 auto r = unpack_ER(right_er, ctx, false);
1913
1914 bool showmuts = (flags & __SHOWMUTS) != 0;
1915
1916 if (showmuts) {
1917 auto lvar_owner = l->borrow_owner();
1918 auto rvar_owner = r->borrow_owner();
1919 std::cout << "[EARL show-muts] ";
1920 std::cout << "<" << (lvar_owner ? lvar_owner->id() : "?") << ">" << l->to_cxxstring();
1921 std::cout << " " << stmt->m_equals->lexeme() << " ";
1922 std::cout << "<" << (rvar_owner ? rvar_owner->id() : "?") << ">" << r->to_cxxstring();
1923 }
1924
1925 switch (stmt->m_equals->type()) {
1926 case TokenType::Equals: {
1927 l->mutate(r.get(), stmt);
1928 if (showmuts)
1929 std::cout << " -> " << l->to_cxxstring() << std::endl;
1930 } break;
1931 case TokenType::Plus_Equals:
1932 case TokenType::Minus_Equals:
1933 case TokenType::Asterisk_Equals:
1934 case TokenType::Forwardslash_Equals:
1935 case TokenType::Percent_Equals:
1936 case TokenType::Backtick_Pipe_Equals:
1937 case TokenType::Backtick_Ampersand_Equals:
1938 case TokenType::Backtick_Caret_Equals: {
1939 l->spec_mutate(stmt->m_equals.get(), r.get(), stmt);
1940 if (showmuts)
1941 std::cout << " -> " << l->to_cxxstring() << std::endl;
1942 } break;
1943 default: {
1944 Err::err_wtok(stmt->m_equals.get());
1945 std::string msg = "invalid mutation operation `"+stmt->m_equals->lexeme()+"`";
1946 throw InterpreterException(msg);
1947 } break;
1948 }
1949 stmt->m_evald = true;
1950 return std::make_shared<earl::value::Void>();
1951}

1990// This function is only used for eval_stmt_foreach() for creating enumerators from a tuple.
1991static void
1992destructure_enumerators(std::vector<std::shared_ptr<Token>> &named_ids,
1993 std::vector<std::shared_ptr<earl::variable::Obj>> &vars,
1994 const std::shared_ptr<earl::value::Obj> &values,
1995 uint32_t attrs,
1996 Expr *expr) {
1997 if (vars.size() == 1)
1998 vars[0] = std::make_shared<earl::variable::Obj>(named_ids[0].get(), values);
1999 else if (values->type() == earl::value::Type::Tuple) {
2000 auto tuple = dynamic_cast<earl::value::Tuple *>(values.get());

2003 Err::err_wexpr(expr);

2014 if ((attrs & static_cast<uint32_t>(Attr::Const)) != 0)
2015 value->set_const();
2016 vars[i] = std::make_shared<earl::variable::Obj>(named_ids[i].get(), tuple->value()[i], attrs);
2017 }
2018 }
2019 else {
2020 Err::err_wexpr(expr);
2021 const std::string msg = "unable to destructure a non-tuple type";
2022 throw InterpreterException(msg);
2023 }
2024}

2057std::shared_ptr<earl::value::Obj>
2058eval_stmt_foreach(StmtForeach *stmt, std::shared_ptr<Ctx> &ctx) {
2059 bool ref = (stmt->m_attrs & static_cast<uint32_t>(Attr::Ref)) != 0;
2060
2061 std::shared_ptr<earl::value::Obj> result = nullptr;
2062 ER expr_er = Interpreter::eval_expr(stmt->m_expr.get(), ctx, ref);
2063 auto expr = unpack_ER(expr_er, ctx, ref);
2064
2065 for (auto &enumer : stmt->m_enumerators) {
2066 const std::string &id = enumer->lexeme();
2067 if (ctx->variable_exists(id)) {
2068 std::string msg = "variable `"+id+"` is already declared";
2069 auto conflict = ctx->variable_get(id);
2070 Err::err_wconflict(enumer.get(), conflict->gettok());
2071 throw InterpreterException(msg);
2072 }
2073 }
2074
2075 std::vector<std::shared_ptr<earl::variable::Obj>> enumerators(stmt->m_enumerators.size(), nullptr);
2076 auto wrapped_iterator = expr->iter_begin(), wrapped_iterator_end = expr->iter_end();
2077
2078 // Will reduce the type of the current iterator and will
2079 // call handle_enumerators() on the enumerators of the foreach loop.
2080 auto iterator_reduce = [&](const auto &it, auto handle_enumerators) {
2081 std::visit([&](const auto &it){
2082 using T = std::decay_t<decltype(it)>;
2083
2084 if constexpr (std::is_same_v<T, earl::value::ListIterator> ||
2085 std::is_same_v<T, earl::value::StrIterator>) {
2086 handle_enumerators(*it);
2087 }
2088 else {
2089 static_assert(std::is_same_v<T, earl::value::DictIntIterator> ||
2090 std::is_same_v<T, earl::value::DictCharIterator> ||
2091 std::is_same_v<T, earl::value::DictFloatIterator> ||
2092 std::is_same_v<T, earl::value::DictStrIterator>);
2093 std::vector<std::shared_ptr<earl::value::Obj>> elements = {};
2094 if constexpr (std::is_same_v<T, earl::value::DictIntIterator>)
2095 elements.push_back(std::make_shared<earl::value::Int>(it->first));
2096 else if constexpr (std::is_same_v<T, earl::value::DictCharIterator>)
2097 elements.push_back(std::make_shared<earl::value::Char>(it->first));
2098 else if constexpr (std::is_same_v<T, earl::value::DictFloatIterator>)
2099 elements.push_back(std::make_shared<earl::value::Float>(it->first));
2100 else if constexpr (std::is_same_v<T, earl::value::DictStrIterator>)
2101 elements.push_back(std::make_shared<earl::value::Str>(it->first));
2102 else {
2103 Err::err_wexpr(stmt->m_expr.get());
2104 const std::string msg = "unknown dictionary iterator type";
2105 throw InterpreterException(msg);
2106 }
2107 elements.push_back(it->second);
2108 auto tuple = std::make_shared<earl::value::Tuple>(elements);
2109 handle_enumerators(tuple);
2110 }
2111 }, it);
2112 };
2113
2114 // The container we are iterating over is empty, nothing to do.
2115 if (wrapped_iterator == wrapped_iterator_end)
2116 goto done;
2117
2118 // Setup starting enumerators
2119 iterator_reduce(wrapped_iterator, [&](auto &tuple){
2120 destructure_enumerators(stmt->m_enumerators, enumerators, tuple, stmt->m_attrs, stmt->m_expr.get());
2121 });
2122 for (size_t i = 0; i < enumerators.size(); ++i)
2123 ctx->variable_add(enumerators[i]);
2124
2125 // Main loop
2126 while (true) {
2127
2128 // Have we enumerated over all elements?
2129 if (std::visit([&](const auto &it) {
2130 return it == std::get<std::decay_t<decltype(it)>>(wrapped_iterator_end);
2131 }, wrapped_iterator)) {
2132 break;
2133 }
