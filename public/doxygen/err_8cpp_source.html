32void
33Err::err_wtok(Token *tok) {
34 if (!tok)
35 return;
36 std::cerr << tok->m_fp << ':' << tok->m_row << ':' << tok->m_col << ":\n";
37 Token *it = tok;
38 while (it && it->type() != TokenType::Semicolon) {
39 std::cerr << it->lexeme();
40 if (it->m_next && it->m_next->type() != TokenType::Semicolon)
41 std::cerr << ' ';
42 it = it->m_next.get();
43 }
44 if (it && it->type() == TokenType::Semicolon)
45 std::cerr << ';';
46 std::cerr << '\n';
47
48 for (int i = 0; i < tok->lexeme().size(); ++i)
49 std::cerr << '^';
50 std::cerr << std::endl;
51}

161err_wterm(ExprTerm *expr, int s) {
162 switch (expr->get_term_type()) {
163 case ExprTermType::Ident: {
164 err_wident(dynamic_cast<ExprIdent *>(expr), s);
165 } break;
166 case ExprTermType::Int_Literal: {
167 err_wintlit(dynamic_cast<ExprIntLit *>(expr), s);
168 } break;
169 case ExprTermType::Str_Literal: {
170 err_wstrlit(dynamic_cast<ExprStrLit *>(expr), s);
171 } break;
172 case ExprTermType::Char_Literal: {
173 err_wcharlit(dynamic_cast<ExprCharLit *>(expr), s);
174 } break;
175 case ExprTermType::Func_Call: {
176 err_wfunccall(dynamic_cast<ExprFuncCall *>(expr), s);
177 } break;
178 case ExprTermType::List_Literal: {
179 err_wlistlit(dynamic_cast<ExprListLit *>(expr), s);

181 case ExprTermType::Range: {
182 err_wrange(dynamic_cast<ExprRange *>(expr), s);

184 case ExprTermType::Slice: {
185 err_wslice(dynamic_cast<ExprSlice *>(expr), s);

187 case ExprTermType::Get: {

324void
325Err::err_wstmt(Stmt *stmt) {
326 if (!stmt) {
327 std::cerr << "[EARL] <unable to display error line at this time>" << std::endl;;
328 return;
329 }
330 switch (stmt->stmt_type()) {
331 case StmtType::Def: assert(false); break;
332 case StmtType::Let: assert(false); break;
333 case StmtType::Block: err_wblockstmt(dynamic_cast<StmtBlock *>(stmt)); break;
334 case StmtType::Mut: err_wmutstmt(dynamic_cast<StmtMut *>(stmt)); break;
335 case StmtType::Stmt_Expr: err_wstmtexpr(dynamic_cast<StmtExpr *>(stmt)); break;
336 case StmtType::If: err_wstmtif(dynamic_cast<StmtIf *>(stmt)); break;
337 case StmtType::Return: err_wstmtreturn(dynamic_cast<StmtReturn *>(stmt)); break;
338 case StmtType::Break: assert(false); break;
339 case StmtType::While: assert(false); break;
340 case StmtType::For: assert(false); break;
341 case StmtType::Foreach: assert(false); break;
342 case StmtType::Import: assert(false); break;
343 case StmtType::Mod: assert(false); break;
344 case StmtType::Class: assert(false); break;
345 case StmtType::Match: assert(false); break;
346 case StmtType::Enum: assert(false); break;
347 default:
348 assert(false && "unreachable");
349 }
