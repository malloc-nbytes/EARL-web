42// To add a new intrinsic, do the following:
43// 1. Add the function signature to this file.
44// 2. In intrinsics.cpp, add the name and the address
45// of the function in the intrinsic_functions map.
46// 3. In intrinsics.cpp, implement the function.
47
49// To add a new member intrinsic, do the following:
50// 1. Add the function signature to this file.
51// 2. In intrinsics.cpp, add the name and the address
52// of the function in the intrinsic_member_functions map.
53// 3. Add the implementation(s) for the appropriate type(s)
54// in primitives/.

56#define __INTR_ARGS_MUSTBE_SIZE(args, sz, fn, expr) \
57 do { \
58 if (args.size() != sz) { \
59 Err::err_wexpr(expr); \
60 std::string __Msg = "function `" fn "` expects "+std::to_string(sz)+" arguments but "+std::to_string(args.size())+" were supplied"; \
61 throw InterpreterException(__Msg); \
62 } \
63 } while (0)
64
65#define __INTR_ARGS_MUSTNOT_BE_0(args, fn, expr) \
66 do { \
67 if (args.size() == 0) { \
68 Err::err_wexpr(expr); \
69 std::string __Msg = "intrinsic `" fn "` expects greater than 0 arguments"; \
70 throw InterpreterException(__Msg); \
71 } \
72 } while (0)
73
74#define __MEMBER_INTR_ARGS_MUSTNOT_BE_0(args, fn, expr) \
75 do { \
76 if (args.size() == 0) { \
77 Err::err_wexpr(expr); \
78 std::string __Msg = "member intrinsic `" fn "` expects greater than 0 arguments"; \
79 throw InterpreterException(__Msg); \
80 } \
81 } while (0)
82
83#define __MEMBER_INTR_ARGS_MUSTBE_GTSIZE(args, sz, fn, expr) \
84 do { \
85 if (args.size() < sz) { \
86 Err::err_wexpr(expr); \
87 std::string __Msg = "member intrinsic `" fn "` expects greater than "+std::to_string(sz)+" arguments but " \
88 +std::to_string(args.size())+" were supplied"; \
89 throw InterpreterException(__Msg); \
90 } \
91 } while (0)
92
93#define __INTR_ARG_MUSTBE_TYPE_COMPAT(arg, ty, loc, fn, expr) \
94 do { \
95 if (!earl::value::type_is_compatable(arg->type(), ty)) { \
96 Err::err_wexpr(expr); \
97 std::string __Msg = "the "+std::to_string(loc)+" argument of function `" fn "` expects type `" \
98 +earl::value::type_to_str(ty)+"` but got `"+earl::value::type_to_str(arg->type())+"`"; \
99 throw InterpreterException(__Msg); \
100 } \
101 } while (0)
102
103#define __INTR_ARG_MUSTBE_TYPE_COMPAT_EXACT(arg, ty, loc, fn, expr) \
104 do { \
105 if (arg->type() != ty) { \
106 Err::err_wexpr(expr); \
107 std::string __Msg = "the "+std::to_string(loc)+" argument of function `" fn "` expects type `" \
108 +earl::value::type_to_str(ty)+"` but got `"+earl::value::type_to_str(arg->type())+"`"; \
109 throw InterpreterException(__Msg); \
110 } \
111 } while (0)
112
113#define __MEMBER_INTR_ARG_MUSTBE_TYPE_COMPAT_OR(arg, ty1, ty2, loc, fn, expr) \
114 do { \
115 if (!earl::value::type_is_compatable(arg->type(), ty1) && !earl::value::type_is_compatable(arg->type(), ty2)) { \
116 Err::err_wexpr(expr); \
117 std::string __Msg = "the "+std::to_string(loc)+" argument of function `" fn "` expects either type `" \
118 +earl::value::type_to_str(ty1)+"` or `"+earl::value::type_to_str(ty2)+"` but got `"+earl::value::type_to_str(arg->type())+"`"; \
119 throw InterpreterException(__Msg); \
120 } \
121 } while (0)
122
123#define __MEMBER_INTR_ARG_MUSTBE_TYPE_COMPAT_OR_LST(arg, tys, loc, fn, expr) \
124 do { \
125 bool __ok = false; \
126 for (auto &ty : tys) { \
127 if (earl::value::type_is_compatable(arg->type(), ty)) \
128 __ok = true; \
129 } \
130 if (!__ok) { \
131 Err::err_wexpr(expr); \
132 std::string __Msg = "the "+std::to_string(loc)+" argument of function `" fn "` expects type-adjacent `" \
133 +earl::value::type_to_str(tys.at(0))+"` but got `"+earl::value::type_to_str(arg->type())+"`"; \
134 throw InterpreterException(__Msg); \
135 } \
136 } while (0)
