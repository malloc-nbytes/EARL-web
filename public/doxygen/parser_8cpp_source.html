40static Attr
41translate_attr(Lexer &lexer) {
42 auto errtok = Parser::parse_expect(lexer, TokenType::At);
43
44 std::shared_ptr<Token> attr = Parser::parse_expect(lexer, TokenType::Ident);
45 if (attr->lexeme() == COMMON_EARLATTR_PUB)
46 return Attr::Pub;
47 if (attr->lexeme() == COMMON_EARLATTR_WORLD)
48 return Attr::World;
49 if (attr->lexeme() == COMMON_EARLATTR_REF)
50 return Attr::Ref;
51 if (attr->lexeme() == COMMON_EARLATTR_CONST)
52 return Attr::Const;
53 if (attr->lexeme() == COMMON_EARLATTR_EXPERIMENTAL)
54 return Attr::Experimental;
55 else {
56 Err::err_wtok(errtok.get());
57 std::string msg = "unknown attribute `" + attr->lexeme() + "`";
58 throw ParserException(msg);
59 }
60}

62static uint32_t
63gather_attrs(Lexer &lexer) {
64 uint32_t attrs = 0;
65 while (lexer.peek(0) && lexer.peek(0)->type() == TokenType::At)
66 attrs |= static_cast<uint32_t>(translate_attr(lexer));
67 return attrs;
68}

70static bool
71is_ty(Lexer &lexer) {
72 if (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Colon
73 && lexer.peek(1) && lexer.peek(1)->type() == TokenType::Ident)
74 return true;
75 return false;
76}

78static std::shared_ptr<__Type>
79get_ty(Lexer &lexer) {
80 std::shared_ptr<Token> main_ty = nullptr;
81 std::optional<std::shared_ptr<Token>> sub_ty = {};
82
83 lexer.discard(); // :
84 main_ty = lexer.next();
85
86 if (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Double_Colon) {
87 lexer.discard();
88 main_ty = Parser::parse_expect(lexer, TokenType::Ident);
89 }
90
91 return std::make_shared<__Type>(main_ty, sub_ty);
92}

94std::shared_ptr<Token>
95Parser::parse_expect(Lexer &lexer, TokenType expected) {
96 std::shared_ptr<Token> tok = lexer.next();
97 if (tok->type() != expected) {
98 Err::err_wtok(tok.get());
99 std::string msg = "expected "
100 + tokentype_to_str(expected)
101 + ", got "
102 + tokentype_to_str(tok->m_type)
103 + " `" + tok->lexeme() + "`";
104 throw ParserException(msg);
105 }
106 return tok;
107}

109std::shared_ptr<Token>
110Parser::parse_expect_keyword(Lexer &lexer, std::string expected) {
111 std::shared_ptr<Token> tok = lexer.next();
112 if (tok->type() != TokenType::Keyword) {
113 Err::err_wtok(tok.get());
114 std::string msg = "expected keyword `"+expected+"`, but got `" + tok->lexeme() + "` which is not a keyword";
115 throw ParserException(msg);
116 }
117 if (tok->lexeme() != expected) {
118 Err::err_wtok(tok.get());
119 std::string msg = "expected keyword `"+expected+"`, but got `" + tok->lexeme() + "` which is not the correct keyword";
120 throw ParserException(msg);
121 }
122 return tok;
123}

125static std::vector<std::shared_ptr<Token>>
126parse_comma_sep_idents_as_toks(Lexer &lexer) {
127 std::vector<std::shared_ptr<Token>> ids = {};
128
129 while (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Ident) {
130 ids.push_back(lexer.next());
131 if (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Comma) {
132 lexer.discard(); // ,
133 }
134 else {
135 if (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Ident) {
136 Err::err_wtok(lexer.peek(0));
137 const std::string msg = "missing comma after identifer";
138 throw ParserException(msg);
139 }
140 }
141 }
142
143 return ids;
144}

146// NOTE: It is up to the calling function to consume the '()' or '[]' or '{}' etc.
147// This makes this function more reusable.
148static std::vector<Expr *>
149parse_comma_sep_exprs(Lexer &lexer, bool &trailing_comma) {
150 std::vector<Expr *> exprs;
151
152 while (1) {
153 // Only needed if no arguments are provided.
154 if (lexer.peek(0) && lexer.peek()->type() == TokenType::Rparen
155 || lexer.peek()->type() == TokenType::Rbrace
156 || lexer.peek()->type() == TokenType::Rbracket) {
157 break;
158 }
159 exprs.push_back(Parser::parse_expr(lexer));
160 if (lexer.peek(0) && lexer.peek()->type() == TokenType::Comma) {
161 (void)Parser::parse_expect(lexer, TokenType::Comma);
162 if (lexer.peek(0) && lexer.peek()->type() == TokenType::Rparen
163 || lexer.peek()->type() == TokenType::Rbrace
164 || lexer.peek()->type() == TokenType::Rbracket)
165 trailing_comma = true;
166 }
167 else
168 break;
169 }
170
171 return exprs;
172}

174static std::optional<std::vector<std::unique_ptr<Expr>>>
175try_parse_funccall(Lexer &lexer) {
176 if (lexer.peek(0) && lexer.peek()->type() == TokenType::Lparen) {
177 (void)Parser::parse_expect(lexer, TokenType::Lparen);
178 bool unused = false;
179 std::vector<Expr *> exprs = parse_comma_sep_exprs(lexer, unused);
180 std::vector<std::unique_ptr<Expr>> unique_exprs = {};
181 for (size_t i = 0; i < exprs.size(); ++i)
182 unique_exprs.push_back(std::unique_ptr<Expr>(exprs[i]));
183 (void)Parser::parse_expect(lexer, TokenType::Rparen);
184 return unique_exprs;
185 }
186
187 return {};
188}

190static std::variant<std::unique_ptr<ExprIdent>, std::unique_ptr<ExprFuncCall>>
191parse_identifier_or_funccall(Lexer &lexer) {
192 auto ident = std::make_unique<ExprIdent>(lexer.next());
193 auto group = try_parse_funccall(lexer);
194
195 if (group.has_value())
196 return std::make_unique<ExprFuncCall>(std::move(ident), std::move(group.value()), ident->m_tok);
197
198 return ident;
199}

201static std::vector<std::pair<std::shared_ptr<Token>, uint32_t>>
202parse_closure_args(Lexer &lexer) {
203 std::vector<std::pair<std::shared_ptr<Token>, uint32_t>> args;
204
205 while (lexer.peek(0) && lexer.peek()->type() != TokenType::Pipe) {
206 uint32_t attr = 0;
207
208 while (lexer.peek(0) && lexer.peek()->type() == TokenType::At) {
209 attr |= static_cast<uint32_t>(translate_attr(lexer));
210 }
211
212 std::shared_ptr<Token> id = Parser::parse_expect(lexer, TokenType::Ident);
213 std::shared_ptr<Token> var = std::move(id);
214 args.push_back(std::make_pair(std::move(var), attr));
215
216 if (lexer.peek(0) && lexer.peek()->type() == TokenType::Comma)
217 lexer.discard();
218 }
219 (void)Parser::parse_expect(lexer, TokenType::Pipe);
220
221 return args;
222}

224static std::vector<std::pair<std::unique_ptr<Expr>, std::unique_ptr<Expr>>>
225parse_set_values(Lexer &lexer) {
226 std::vector<std::pair<std::unique_ptr<Expr>, std::unique_ptr<Expr>>> values = {};
227 while (lexer.peek(0) && lexer.peek(0)->type() != TokenType::Rbrace) {
228 Expr *key = Parser::parse_expr(lexer, /*fail_on=*/':');
229 Parser::parse_expect(lexer, TokenType::Colon);
230 Expr *value = Parser::parse_expr(lexer);
231 values.push_back(std::make_pair(std::unique_ptr<Expr>(key), std::unique_ptr<Expr>(value)));
232 if (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Comma)
233 lexer.discard(); // ,
234 else {
235 Parser::parse_expect(lexer, TokenType::Rbrace);
236 break;
237 }
238 }
239 return values;
240}

242static Expr *
243parse_primary_expr(Lexer &lexer, char fail_on = '\0') {
244 Token *tok = nullptr;
245 Expr *left = nullptr;
246 Expr *right = nullptr;
247
248 // Unary expressions
249 while (lexer.peek(0) && (lexer.peek()->type() == TokenType::Minus
250 || lexer.peek()->type() == TokenType::Bang
251 || lexer.peek()->type() == TokenType::Backtick_Tilde)) {
252 std::shared_ptr<Token> op = lexer.next();
253 Expr *operand = parse_primary_expr(lexer, fail_on);
254 left = new ExprUnary(std::move(op), std::unique_ptr<Expr>(operand));
255 }
256
257 while (1) {
258 if (!lexer.peek(0))
259 return left;
260 switch (lexer.peek()->type()) {
261 case TokenType::Ident: {
262 left = new ExprIdent(lexer.next());
263 } break;
264 case TokenType::Lbrace: {
265 if (fail_on == '{')
266 return left;
267 auto tok = lexer.next();
268 auto values = parse_set_values(lexer);
269 left = new ExprDict(std::move(values), tok);
270 } break;
271 case TokenType::Lparen: {
272 auto tok = lexer.next(); // (
273 bool trailing_comma = false;
274 std::vector<Expr *> tuple = parse_comma_sep_exprs(lexer, trailing_comma);
275 (void)Parser::parse_expect(lexer, TokenType::Rparen);
276 // Function call
277 if (left) {
278 std::vector<std::unique_ptr<Expr>> unique_tuple = {};
279 for (size_t i = 0; i < tuple.size(); ++i)
280 unique_tuple.push_back(std::unique_ptr<Expr>(tuple[i]));
281 left = new ExprFuncCall(std::unique_ptr<Expr>(left), std::move(unique_tuple), tok);
282 }
283 // Tuple
284 else if (tuple.size() > 1 || trailing_comma) {
285 std::vector<std::unique_ptr<Expr>> unique_tuple = {};
286 for (size_t i = 0; i < tuple.size(); ++i)
287 unique_tuple.push_back(std::unique_ptr<Expr>(tuple[i]));
288 left = new ExprTuple(std::move(unique_tuple), tok);
289 }
290 // Math
291 else
292 left = std::move(tuple[0]);
293 } break;
294 case TokenType::Period: {
295 if (!left) {
296 Err::err_wtok(lexer.peek(0));
297 const std::string msg = "cannot use dot notation `.` with no lhs";
298 throw ParserException(msg);
299 }
300 auto tok = lexer.next();
301 left = new ExprGet(std::unique_ptr<Expr>(left), parse_identifier_or_funccall(lexer), tok);
302 } break;
303 case TokenType::Intlit: {
304 left = new ExprIntLit(lexer.next());
305 } break;
306 case TokenType::Floatlit: {
307 left = new ExprFloatLit(lexer.next());
308 } break;
309 case TokenType::Strlit: {
310 if (left && left->get_type() == ExprType::Term) {
311 auto left_term = dynamic_cast<ExprTerm *>(left);
312 if (left_term->get_term_type() == ExprTermType::Ident) {
313 auto left_ident = dynamic_cast<ExprIdent *>(left_term);
314 if (left_ident->m_tok->lexeme() == "f")
315 left = new ExprFStr(lexer.next());
316 else
317 goto not_fstr;
318 }
319 else
320 goto not_fstr;
321 }
322 else {
323 not_fstr:
324 left = new ExprStrLit(lexer.next());
325 }
326 } break;
327 case TokenType::Charlit: {
328 left = new ExprCharLit(lexer.next());
329 } break;
330 case TokenType::Lbracket: {
331 if (left) {
332 auto tok = lexer.next(); // [
333 Expr *idx = Parser::parse_expr(lexer);
334 if (!idx) {
335 Err::err_wtok(tok.get());
336 const std::string msg = "list access requires an index";
337 throw ParserException(msg);
338 }
339 (void)Parser::parse_expect(lexer, TokenType::Rbracket);
340 left = new ExprArrayAccess(std::unique_ptr<Expr>(left), std::unique_ptr<Expr>(idx), tok);
341 }
342 else {
343 auto tok = lexer.next(); // [
344 bool unused = false;
345 std::vector<Expr *> lst = parse_comma_sep_exprs(lexer, unused);
346 std::vector<std::unique_ptr<Expr>> unique_lst = {};
347 for (size_t i = 0; i < lst.size(); ++i)
348 unique_lst.push_back(std::unique_ptr<Expr>(lst[i]));
349 (void)Parser::parse_expect(lexer, TokenType::Rbracket);
350 left = new ExprListLit(std::move(unique_lst), tok);
351 }
352 } break;
353 case TokenType::Double_Colon: {
354 auto tok = lexer.next(); // ::
355
356 if (left->get_type() == ExprType::Term) {
357 auto _left = dynamic_cast<ExprTerm *>(left);
358 if (_left->get_term_type() == ExprTermType::Ident)
359 left = new ExprModAccess(std::unique_ptr<ExprIdent>(dynamic_cast<ExprIdent *>(_left)),
360 parse_identifier_or_funccall(lexer),
361 tok);
362 else {
363 std::string msg = "Module getters must be of term type identifier";
364 throw ParserException(msg);
365 }
366 }
367 else {
368 std::string msg = "Module getters must be of term type identifier";
369 throw ParserException(msg);
370 }
371
372 } break;
373 case TokenType::Pipe: {
374 // Workaround for when parsing a `match` statement since
375 // the individual branches can have `|` in arm. We want to
376 // fail here and let the callees handle it.
377 if (fail_on == '|')
378 return left;
379 auto tok = lexer.next(); // |
380 std::vector<std::pair<std::shared_ptr<Token>, uint32_t>> args = parse_closure_args(lexer);
381 auto block = Parser::parse_stmt_block(lexer);
382 return new ExprClosure(std::move(args), std::move(block), tok);
383 }
384 case TokenType::Keyword: {
385 if (lexer.peek(0) && lexer.peek()->lexeme() == COMMON_EARLKW_TO)
386 return left;
387 if (lexer.peek(0) && lexer.peek()->lexeme() == COMMON_EARLKW_WHEN)
388 return left;
389
390 std::shared_ptr<Token> kw = lexer.next();
391 if (kw->lexeme() == COMMON_EARLKW_TRUE) {
392 return new ExprBool(std::move(kw), true);
393 }
394 else if (kw->lexeme() == COMMON_EARLKW_FALSE) {
395 return new ExprBool(std::move(kw), false);
396 }
397 else if (kw->lexeme() == COMMON_EARLKW_NONE) {
398 return new ExprNone(std::move(kw));
399 }
400 else {
401 Err::err_wtok(kw.get());
402 std::string msg = "invalid keyword `" + kw->lexeme() + "` while parsing primary expression";
403 throw ParserException(msg);
404 }
405 } break;
406 default: {
407 return left;
408 } break;
409 }
410 }
411 return left;
412}

414static Expr *
415parse_power_expr(Lexer &lexer, char fail_on='\0') {
416 Expr *lhs = parse_primary_expr(lexer, fail_on);
417 Token *cur = lexer.peek();
418 while (cur && (cur->type() == TokenType::Double_Asterisk)) {
419 std::shared_ptr<Token> op = lexer.next();
420 Expr *rhs = parse_primary_expr(lexer, fail_on);
421 lhs = new ExprBinary(std::unique_ptr<Expr>(lhs),
422 std::move(op),
423 std::unique_ptr<Expr>(rhs));
424 cur = lexer.peek();
425 }
426 return lhs;
427}

429static Expr *
430parse_multiplicative_expr(Lexer &lexer, char fail_on = '\0') {
431 Expr *lhs = parse_power_expr(lexer, fail_on);
432 Token *cur = lexer.peek();

434 || cur->type() == TokenType::Forwardslash
435 || cur->type() == TokenType::Percent)) {

437 Expr *rhs = parse_power_expr(lexer, fail_on);
438 lhs = new ExprBinary(std::unique_ptr<Expr>(lhs),
439 std::move(op),
440 std::unique_ptr<Expr>(rhs));
441 cur = lexer.peek();
442 }
443 return lhs;

446static Expr *
447parse_additive_expr(Lexer &lexer, char fail_on = '\0') {
448 Expr *lhs = parse_multiplicative_expr(lexer, fail_on);
449 Token *cur = lexer.peek();
450 while (cur && (cur->type() == TokenType::Plus
451 || cur->type() == TokenType::Minus)) {
452 std::shared_ptr<Token> op = lexer.next();
453 Expr *rhs = parse_multiplicative_expr(lexer, fail_on);
454 lhs = new ExprBinary(std::unique_ptr<Expr>(lhs),
455 std::move(op),
456 std::unique_ptr<Expr>(rhs));
457 cur = lexer.peek();
458 }
459 return lhs;
460}

462static Expr *
463parse_equalitative_expr(Lexer &lexer, char fail_on = '\0') {
464 Expr *lhs = parse_additive_expr(lexer, fail_on);
465 Token *cur = lexer.peek();
466 while (cur && (cur->type() == TokenType::Double_Equals
467 || cur->type() == TokenType::Greaterthan_Equals
468 || cur->type() == TokenType::Greaterthan
469 || cur->type() == TokenType::Lessthan_Equals
470 || cur->type() == TokenType::Lessthan
471 || cur->type() == TokenType::Bang_Equals)) {
472 std::shared_ptr<Token> op = lexer.next();
473 Expr *rhs = parse_additive_expr(lexer, fail_on);
474 lhs = new ExprBinary(std::unique_ptr<Expr>(lhs),
475 std::move(op),
476 std::unique_ptr<Expr>(rhs));
477 cur = lexer.peek();
478 }
479 return lhs;
480}

517static Expr *
518parse_slice_expr(Lexer &lexer, char fail_on = '\0') {
519 Expr *lhs = parse_bitwise_expr(lexer, fail_on);
520 Token *cur = lexer.peek();
521 while (fail_on != ':' && cur && (cur->type() == TokenType::Colon)) {
522 auto tok = lexer.next(); // :
523 Expr *rhs = Parser::parse_expr(lexer, fail_on);
524 std::optional<std::unique_ptr<Expr>> l = {}, r = {};
525 if (lhs)
526 l = std::unique_ptr<Expr>(lhs);
527 if (rhs)
528 r = std::unique_ptr<Expr>(rhs);
529 lhs = new ExprSlice(std::move(l), std::move(r), tok);
530 cur = lexer.peek();
531 }
532 return lhs;
533}

535static Expr *
536parse_range_expr(Lexer &lexer, char fail_on = '\0') {
537 Expr *lhs = parse_slice_expr(lexer, fail_on);
538 Token *cur = lexer.peek();
539 while (cur && (cur->type() == TokenType::Double_Period)) {
540 auto tok = lexer.next(); // ..
541 Expr *rhs = nullptr;
542 if (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Equals) {
543 lexer.discard(); // =
544 rhs = Parser::parse_expr(lexer, /*fail_on=*/'{');
545 lhs = new ExprRange(std::unique_ptr<Expr>(lhs), std::unique_ptr<Expr>(rhs), true, tok);
546 }
547 else {
548 rhs = Parser::parse_expr(lexer, /*fail_on=*/'{');
549 if (!rhs) {
550 Err::err_wtok(lexer.peek(0));
551 std::string msg = "cannot use a range where the end expression is empty";
552 throw ParserException(msg);
553 }
554 lhs = new ExprRange(std::unique_ptr<Expr>(lhs), std::unique_ptr<Expr>(rhs), false, tok);
555 }
556
557 cur = lexer.peek();
558 }
559 return lhs;
560}

562Expr *
563Parser::parse_expr(Lexer &lexer, char fail_on) {
564 return parse_range_expr(lexer, fail_on);

567std::unique_ptr<StmtMut>
568Parser::parse_stmt_mut(Lexer &lexer) {
569 Expr *left = Parser::parse_expr(lexer);
570 auto tok = lexer.next(); // =, +=, -=, etc
571 Expr *right = Parser::parse_expr(lexer);
572 (void)parse_expect(lexer, TokenType::Semicolon);
573 return std::make_unique<StmtMut>(std::unique_ptr<Expr>(left),
574 std::unique_ptr<Expr>(right),
575 std::move(tok));
576}

578std::unique_ptr<StmtIf>
579Parser::parse_stmt_if(Lexer &lexer) {
580 (void)Parser::parse_expect_keyword(lexer, COMMON_EARLKW_IF);
581
582 Expr *expr = Parser::parse_expr(lexer, /*fail_on=*/'{');
583 std::unique_ptr<StmtBlock> block = Parser::parse_stmt_block(lexer);
584
585 // Handle the `else if` or `else` blocks if applicable
586 std::optional<std::unique_ptr<StmtBlock>> else_ = {};
587 Token *tok1 = lexer.peek();
588 Token *tok2 = lexer.peek(1);
589
590 bool tok1_else = tok1->type() == TokenType::Keyword && tok1->lexeme() == COMMON_EARLKW_ELSE;
591 bool tok2_if = tok2 && (tok2->type() == TokenType::Keyword && tok2->lexeme() == COMMON_EARLKW_IF);
592
593 if (tok1_else && tok2_if) {
594 lexer.discard();
595 std::vector<std::unique_ptr<Stmt>> tmp;
596 std::unique_ptr<StmtIf> nested_if = parse_stmt_if(lexer);
597 tmp.push_back(std::move(nested_if));
598 else_ = std::make_unique<StmtBlock>(std::move(tmp));
599 }
600 else if (tok1_else) {
601 Parser::parse_expect_keyword(lexer, COMMON_EARLKW_ELSE);
602 else_ = parse_stmt_block(lexer);
603 }
604
605 return std::make_unique<StmtIf>(std::unique_ptr<Expr>(expr),
606 std::move(block),
607 std::move(else_));

610std::unique_ptr<StmtLet>
611Parser::parse_stmt_let(Lexer &lexer, uint32_t attrs, std::vector<std::string> info) {
612 auto errtok = Parser::parse_expect_keyword(lexer, COMMON_EARLKW_LET);
613
614 std::vector<std::shared_ptr<Token>> ids = {parse_expect(lexer, TokenType::Ident)};
615 std::vector<std::shared_ptr<__Type>> tys = {};
616
617 if (is_ty(lexer)) {
618 tys.push_back(get_ty(lexer));
619 }
620
621 while (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Comma) {
622 lexer.discard(); // ,
623 ids.push_back(parse_expect(lexer, TokenType::Ident));
624 if (is_ty(lexer))
625 tys.push_back(get_ty(lexer));
626 }
627
628 if (tys.size() > 0 && ids.size() != tys.size()) {
629 Err::err_wtok(errtok.get());
630 const std::string msg = "when declaring multiple variables with explicit types, all variables must have a type";
631 throw ParserException(msg);
632 }
633
634 (void)parse_expect(lexer, TokenType::Equals);
635 Expr *expr = Parser::parse_expr(lexer);
636
637 if (!expr) {
638 Err::err_wtok(lexer.peek());
639 std::string msg = "invalid token `" + lexer.peek()->lexeme() + "` while parsing primary expression";
640 throw ParserException(msg);
641 }
642
643 (void)parse_expect(lexer, TokenType::Semicolon);
644 return std::make_unique<StmtLet>(std::move(ids), std::move(tys), std::unique_ptr<Expr>(expr), attrs, std::move(info));
645}

672 std::vector<std::pair<std::pair<std::shared_ptr<Token>, std::optional<std::shared_ptr<__Type>>>, uint32_t>>

692 lexer.discard();
693 }
694 (void)Parser::parse_expect(lexer, TokenType::Rparen);
695
696 return args;
697}

841std::unique_ptr<StmtClass>
842parse_stmt_class(Lexer &lexer, uint32_t attrs, std::vector<std::string> info) {
843 (void)Parser::parse_expect_keyword(lexer, COMMON_EARLKW_CLASS);
844
845 std::shared_ptr<Token> class_id = Parser::parse_expect(lexer, TokenType::Ident);
846
847 std::vector<std::unique_ptr<StmtLet>> members;
848 std::vector<std::unique_ptr<StmtDef>> methods;
849
850 auto constructor_args = parse_stmt_class_constructor_arguments(lexer);
851
852 (void)Parser::parse_expect(lexer, TokenType::Lbrace);
853
854 uint32_t inclass_attrs = 0;
855 while (lexer.peek(0) && lexer.peek()->type() != TokenType::Rbrace) {
856
857 std::vector<std::string> inclass_info = {};
858 inclass_attrs = 0;
859 do {
860 if (lexer.peek(0) && lexer.peek()->type() == TokenType::At)
861 inclass_attrs |= static_cast<uint32_t>(translate_attr(lexer));
862 else if (lexer.peek(0) && lexer.peek()->type() == TokenType::Info)
863 inclass_info.push_back(lexer.next()->lexeme());
864 else
865 break;
866 } while (inclass_attrs != 0 || inclass_info.size() != 0);
867
868 Token *tok = lexer.peek();
869 assert(tok);
870
871 switch (tok->type()) {
872 case TokenType::Keyword: {
873 if (tok->lexeme() == COMMON_EARLKW_LET)
874 members.push_back(Parser::parse_stmt_let(lexer, inclass_attrs, std::move(inclass_info)));
875 else if (tok->lexeme() == COMMON_EARLKW_FN)
876 methods.push_back(Parser::parse_stmt_def(lexer, inclass_attrs, std::move(inclass_info)));
877 else {
878 Err::err_wtok(tok);
879 std::string msg = "invalid keyword specifier `" + tok->lexeme() + "` in class declaration";
880 throw ParserException(msg);
881 }
882 } break;
883 default: {
884 Err::err_wtok(tok);
885 std::string msg = "invalid token type (" + std::to_string((int)tok->type()) + ") in class declaration";
886 throw ParserException(msg);
887 } break;
888 }
889 }
890
891 (void)Parser::parse_expect(lexer, TokenType::Rbrace);
892
893 return std::make_unique<StmtClass>(std::move(class_id),
894 attrs,
895 std::move(constructor_args),
896 std::move(members),
897 std::move(methods),
898 std::move(info));
899}

901static std::unique_ptr<StmtMatch::Branch>
902parse_branch(Lexer &lexer) {
903 std::vector<std::unique_ptr<Expr>> exprs = {};
904 std::optional<std::unique_ptr<Expr>> when = {};
905 std::unique_ptr<StmtBlock> block = nullptr;
906
907 while (1) {
908 exprs.push_back(std::unique_ptr<Expr>(Parser::parse_expr(lexer, /*fail_on=*/'|')));
909 if (lexer.peek(0) && lexer.peek()->type() == TokenType::Pipe)
910 lexer.discard();
911 else
912 break;
913 }
914
915 if (lexer.peek(0) && lexer.peek()->lexeme() == COMMON_EARLKW_WHEN) {
916 (void)Parser::parse_expect_keyword(lexer, COMMON_EARLKW_WHEN);
917 when = std::unique_ptr<Expr>(Parser::parse_expr(lexer));
918 }
919
920 (void)Parser::parse_expect(lexer, TokenType::RightArrow);
921
922 block = Parser::parse_stmt_block(lexer);
923
924 return std::make_unique<StmtMatch::Branch>(std::move(exprs), std::move(when), std::move(block));
925}

928parse_branches(Lexer &lexer) {
929 std::vector<std::unique_ptr<StmtMatch::Branch>> branches;
930 while (lexer.peek(0) && lexer.peek()->type() != TokenType::Rbrace)
931 branches.push_back(parse_branch(lexer));
932 return branches;
933}

1007
1008std::unique_ptr<Stmt>
1009Parser::parse_stmt(Lexer &lexer) {
1010
1011 uint32_t attrs = 0;
1012 std::vector<std::string> info = {};
1013
1014 do {
1015 Token *tok = lexer.peek();
1016
1017 switch (tok->type()) {
1018 case TokenType::Dollarsign: {
1019 auto bash_stmt = parse_stmt_bash(lexer);
1020
1021 if (lexer.peek(0) && lexer.peek(0)->type() == TokenType::Pipe_Greaterthan) {
1022 lexer.discard(); // |>
1023 if (lexer.peek(0) &&
1024 lexer.peek(0)->type() == TokenType::Keyword &&
1025 lexer.peek(0)->m_lexeme == COMMON_EARLKW_LET) {
1026 lexer.discard(); // let
1027 auto tok = Parser::parse_expect(lexer, TokenType::Ident);
1028 (void)Parser::parse_expect(lexer, TokenType::Semicolon);
1029 return std::make_unique<StmtPipe>(std::move(bash_stmt), std::move(tok), attrs, std::move(info));
1030 }
1031 auto expr = Parser::parse_expr(lexer);
1032 (void)Parser::parse_expect(lexer, TokenType::Semicolon);
1033 return std::make_unique<StmtPipe>(std::move(bash_stmt), std::unique_ptr<Expr>(expr), attrs, std::move(info));
1034 }
1035
1036 (void)Parser::parse_expect(lexer, TokenType::Semicolon);
1037 return bash_stmt;
1038 } break;
1039 case TokenType::Keyword: {
1040 if (tok->lexeme() == COMMON_EARLKW_FN)
1041 return parse_stmt_def(lexer, attrs, std::move(info));
1042 if (tok->lexeme() == COMMON_EARLKW_ENUM)
1043 return parse_stmt_enum(lexer, attrs, std::move(info));
1044 if (tok->lexeme() == COMMON_EARLKW_CLASS)
1045 return parse_stmt_class(lexer, attrs, std::move(info));
1046 if (tok->lexeme() == COMMON_EARLKW_LET)
1047 return parse_stmt_let(lexer, attrs, std::move(info));
1048
1049 if (info.size() > 0) {
1050 Err::err_wtok(tok);
1051 const std::string msg = "Info statements are only available for functions, enums, and classes";
1052 throw ParserException(msg);
1053 }
1054
1055 if (tok->lexeme() == COMMON_EARLKW_IF)
1056 return parse_stmt_if(lexer);
1057 if (tok->lexeme() == COMMON_EARLKW_RETURN)
1058 return parse_stmt_return(lexer);
1059 if (tok->lexeme() == COMMON_EARLKW_BREAK)
1060 return parse_stmt_break(lexer);
1061 if (tok->lexeme() == COMMON_EARLKW_WHILE)
1062 return parse_stmt_while(lexer);
1063 if (tok->lexeme() == COMMON_EARLKW_FOREACH)
1064 return parse_stmt_foreach(lexer);
1065 if (tok->lexeme() == COMMON_EARLKW_FOR)
1066 return parse_stmt_for(lexer);
1067 if (tok->lexeme() == COMMON_EARLKW_IMPORT)
1068 return parse_stmt_import(lexer);
1069 if (tok->lexeme() == COMMON_EARLKW_MODULE)
1070 return parse_stmt_mod(lexer);
1071 if (tok->lexeme() == COMMON_EARLKW_MATCH)
1072 return parse_stmt_match(lexer);
1073 if (tok->lexeme() == COMMON_EARLKW_CONTINUE)
1074 return parse_stmt_continue(lexer);
1075 if (tok->lexeme() == COMMON_EARLKW_LOOP)
1076 return parse_stmt_loop(lexer);
1077 if (tok->lexeme() == COMMON_EARLKW_NONE
1078 || tok->lexeme() == COMMON_EARLKW_TRUE
1079 || tok->lexeme() == COMMON_EARLKW_FALSE)
1080 return parse_stmt_expr(lexer);
1081 Err::err_wtok(tok);
1082 std::string msg = "invalid keyword `" + tok->lexeme() + "`";
1083 throw ParserException(msg);
1084 } break;
1085 case TokenType::Ident: {
1086 if (info.size() > 0) {
1087 Err::err_wtok(tok);
1088 const std::string msg = "Info statements are only available for variables, functions, enums, and classes";
1089 throw ParserException(msg);
1090 }
1091
1092 // Keeping track of parens to fix an equals sign being in a closure
1093 // that is defined inside of a function call.
1094 int paren = 0;
1095 for (size_t i = 0; lexer.peek(i) && lexer.peek(i) && lexer.peek(i)->type() != TokenType::Semicolon; ++i) {
1096 if (lexer.peek(i)->type() == TokenType::Lparen)
1097 ++paren;
1098 if (lexer.peek(i)->type() == TokenType::Rparen)
1099 --paren;
1100 if ((lexer.peek(i)->type() == TokenType::Equals
1101 || lexer.peek(i)->type() == TokenType::Plus_Equals
1102 || lexer.peek(i)->type() == TokenType::Minus_Equals
1103 || lexer.peek(i)->type() == TokenType::Asterisk_Equals
1104 || lexer.peek(i)->type() == TokenType::Forwardslash_Equals
1105 || lexer.peek(i)->type() == TokenType::Percent_Equals
1106 || lexer.peek(i)->type() == TokenType::Backtick_Pipe_Equals
1107 || lexer.peek(i)->type() == TokenType::Backtick_Ampersand_Equals
1108 || lexer.peek(i)->type() == TokenType::Backtick_Caret_Equals)
1109 && paren == 0)
1110 return parse_stmt_mut(lexer);
1111 }
1112 return parse_stmt_expr(lexer);
1113 } break;
1114 case TokenType::At: {
1115 attrs |= static_cast<uint32_t>(translate_attr(lexer));
1116 } break;
1117 case TokenType::Info: {
1118 info.push_back(lexer.next()->lexeme());
1119 } break;
1120 case TokenType::Semicolon: {
1121 Err::err_wtok(tok);
1122 std::string msg = "expected a statement or expression but got excess semicolon(s)";
1123 throw ParserException(msg);
1124 } break;
1125 default: {
1126 if (info.size() > 0) {
1127 Err::err_wtok(tok);
1128 const std::string msg = "Info statements are only available for variables, functions, enums, and classes";
1129 throw ParserException(msg);
1130 }
1131 return parse_stmt_expr(lexer);
1132 }
1133 }
1134 } while (attrs != 0 || info.size() != 0);
1135
1136 std::string msg = "A serious internal error has ocured and has gotten to an unreachable case. Something is very wrong";
1137 throw ParserException(msg);
1138 return nullptr;
1139}

1141std::unique_ptr<Program>
1142Parser::parse_program(Lexer &lexer, const std::string filepath, std::string from) {
1143 if ((flags & __VERBOSE) != 0)
1144 std::cout << "[EARL] parsing file " << filepath << std::endl;
1145
