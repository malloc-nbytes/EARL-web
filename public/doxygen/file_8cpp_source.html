38File::File(std::shared_ptr<Str> fp, std::shared_ptr<Str> mode, std::fstream stream)
39 : m_fp(fp), m_mode(mode),
40 m_stream(std::move(stream)), m_open(false), m_mode_actual(0) {
41 const std::string &literal = m_mode->value();
42 for (char c : literal) {
43 switch (c) {
44 case 'w': m_mode_actual |= static_cast<uint32_t>(Mode::Write); break;
45 case 'r': m_mode_actual |= static_cast<uint32_t>(Mode::Read); break;
46 case 'b': m_mode_actual |= static_cast<uint32_t>(Mode::Binary); break;
47 default:
48 std::string msg = "unknown mode `"+std::to_string(c)+"` for file type";
49 throw InterpreterException(msg);
50 break;
51 }
52 }
53}

105void
106File::write(std::shared_ptr<Obj> value) {
107 if (!m_open) {
108 std::string msg = "file is not open";
109 throw InterpreterException(msg);
110 }
111
112 if ((m_mode_actual & static_cast<uint32_t>(Mode::Write)) == 0) {
113 std::string msg = "file is not open for writing";
114 throw InterpreterException(msg);
115 }
116
117 switch (value->type()) {
118 case Type::Int: {
119 auto _int = dynamic_cast<Int *>(value.get());
120 m_stream << _int->value();
121 } break;
122 case Type::Char: {
123 auto _char = dynamic_cast<Char *>(value.get());
124 m_stream << _char->value();
125 } break;
126 case Type::Str: {
127 auto str = dynamic_cast<Str *>(value.get());
128 m_stream << str->value();
129 } break;
130 default: {
131 std::string msg = "cannot write `"+type_to_str(value->type())+"` type to a file";
132 throw InterpreterException(msg);
133 } break;
134 }
135}
